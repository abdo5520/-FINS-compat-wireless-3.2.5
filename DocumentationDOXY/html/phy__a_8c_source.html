00267 {

00279                 if (phy->type == B43_PHYTYPE_A)
00280                         b43_phy_mask(dev, B43_PHY_OFDM(0x1B), ~0x1000);

00422                 offset &= ~B43_PHYROUTE;
00423                 offset |= B43_PHYROUTE_BASE;

00430                        "0x%04X on A-PHY\n", offset);
