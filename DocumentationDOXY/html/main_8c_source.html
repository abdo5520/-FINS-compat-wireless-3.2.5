00849 /* The ucode will use phase1 key with TEK key to decrypt rx packets.
00850  * When a packet is received, the iv32 is checked.
00851  * - if it doesn't the packet is returned without modification (and software
00852  *   decryption can be done). That's what happen when iv16 wrap.
00853  * - if it does, the rc4 key is computed, and decryption is tried.
00854  *   Either it will success and B43_RX_MAC_DEC is returned,
00855  *   either it fails and B43_RX_MAC_DEC|B43_RX_MAC_DECERR is returned
00856  *   and the packet is not usable (it got modified by the ucode).
00857  * So in order to never have B43_RX_MAC_DECERR, we should provide
00858  * a iv32 and phase1key that match. Because we drop packets in case of
00859  * B43_RX_MAC_DECERR, if we have a correct iv32 but a wrong phase1key, all
00860  * packets will be lost without higher layer knowing (ie no resync possible
00861  * until next wrap).
00862  *
00863  * NOTE : this should support 50 key like RCMTA because
00864  * (B43_SHM_SH_KEYIDXBLOCK - B43_SHM_SH_TKIPTSCTTAK)/14 = 50
00865  */
00866 static void rx_tkip_phase1_write(struct b43_wldev *dev, u8 index, u32 iv32,
00867                 u16 *phase1key)
00868 {
00869         unsigned int i;
00870         u32 offset;
00871         u8 pairwise_keys_start = B43_NR_GROUP_KEYS * 2;
00872 
00873         if (!modparam_hwtkip)
00874                 return;
00875 
00876         if (b43_new_kidx_api(dev))
00877                 pairwise_keys_start = B43_NR_GROUP_KEYS;
00878 
00879         B43_WARN_ON(index < pairwise_keys_start);
00880         /* We have four default TX keys and possibly four default RX keys.
00881          * Physical mac 0 is mapped to physical key 4 or 8, depending
00882          * on the firmware version.
00883          * So we must adjust the index here.
00884          */
00885         index -= pairwise_keys_start;
00886         B43_WARN_ON(index >= B43_NR_PAIRWISE_KEYS);
00887 
00888         if (b43_debug(dev, B43_DBG_KEYS)) {
00889                 b43dbg(dev->wl, "rx_tkip_phase1_write : idx 0x%x, iv32 0x%x\n",
00890                                 index, iv32);
00891         }
00892         /* Write the key to the  RX tkip shared mem */
00893         offset = B43_SHM_SH_TKIPTSCTTAK + index * (10 + 4);
00894         for (i = 0; i < 10; i += 2) {
00895                 b43_shm_write16(dev, B43_SHM_SHARED, offset + i,
00896                                 phase1key ? phase1key[i / 2] : 0);
00897         }
00898         b43_shm_write16(dev, B43_SHM_SHARED, offset + i, iv32);
00899         b43_shm_write16(dev, B43_SHM_SHARED, offset + i + 2, iv32 >> 16);
00900 }

00930 static void do_key_write(struct b43_wldev *dev,
00931                          u8 index, u8 algorithm,
00932                          const u8 *key, size_t key_len, const u8 *mac_addr)
00933 {
00934         u8 buf[B43_SEC_KEYSIZE] = { 0, };
00935         u8 pairwise_keys_start = B43_NR_GROUP_KEYS * 2;
00936 
00937         if (b43_new_kidx_api(dev))
00938                 pairwise_keys_start = B43_NR_GROUP_KEYS;
00939 
00940         B43_WARN_ON(index >= ARRAY_SIZE(dev->key));
00941         B43_WARN_ON(key_len > B43_SEC_KEYSIZE);
00942 
00943         if (index >= pairwise_keys_start)
00944                 keymac_write(dev, index, NULL); /* First zero out mac. */
00945         if (algorithm == B43_SEC_ALGO_TKIP) {
00946                 /*
00947                  * We should provide an initial iv32, phase1key pair.
00948                  * We could start with iv32=0 and compute the corresponding
00949                  * phase1key, but this means calling ieee80211_get_tkip_key
00950                  * with a fake skb (or export other tkip function).
00951                  * Because we are lazy we hope iv32 won't start with
00952                  * 0xffffffff and let's b43_op_update_tkip_key provide a
00953                  * correct pair.
00954                  */
00955                 rx_tkip_phase1_write(dev, index, 0xffffffff, (u16*)buf);
00956         } else if (index >= pairwise_keys_start) /* clear it */
00957                 rx_tkip_phase1_write(dev, index, 0, NULL);
00958         if (key)
00959                 memcpy(buf, key, key_len);
00960         key_write(dev, index, algorithm, buf);
00961         if (index >= pairwise_keys_start)
00962                 keymac_write(dev, index, mac_addr);
00963 
00964         dev->key[index].algorithm = algorithm;
00965 }

00967 static int b43_key_write(struct b43_wldev *dev,
00968                          int index, u8 algorithm,
00969                          const u8 *key, size_t key_len,
00970                          const u8 *mac_addr,
00971                          struct ieee80211_key_conf *keyconf)
00972 {
00973         int i;
00974         int pairwise_keys_start;
00975 
00976         /* For ALG_TKIP the key is encoded as a 256-bit (32 byte) data block:
00977          *      - Temporal Encryption Key (128 bits)
00978          *      - Temporal Authenticator Tx MIC Key (64 bits)
00979          *      - Temporal Authenticator Rx MIC Key (64 bits)
00980          *
00981          *      Hardware only store TEK
00982          */
00983         if (algorithm == B43_SEC_ALGO_TKIP && key_len == 32)
00984                 key_len = 16;
00985         if (key_len > B43_SEC_KEYSIZE)
00986                 return -EINVAL;
00987         for (i = 0; i < ARRAY_SIZE(dev->key); i++) {
00988                 /* Check that we don't already have this key. */
00989                 B43_WARN_ON(dev->key[i].keyconf == keyconf);
00990         }
00991         if (index < 0) {
00992                 /* Pairwise key. Get an empty slot for the key. */
00993                 if (b43_new_kidx_api(dev))
00994                         pairwise_keys_start = B43_NR_GROUP_KEYS;
00995                 else
00996                         pairwise_keys_start = B43_NR_GROUP_KEYS * 2;
00997                 for (i = pairwise_keys_start;
00998                      i < pairwise_keys_start + B43_NR_PAIRWISE_KEYS;
00999                      i++) {
01000                         B43_WARN_ON(i >= ARRAY_SIZE(dev->key));
01001                         if (!dev->key[i].keyconf) {
01002                                 /* found empty */
01003                                 index = i;
01004                                 break;
01005                         }
01006                 }
01007                 if (index < 0) {
01008                         b43warn(dev->wl, "Out of hardware key memory\n");
01009                         return -ENOSPC;
01010                 }
01011         } else
01012                 B43_WARN_ON(index > 3);
01013 
01014         do_key_write(dev, index, algorithm, key, key_len, mac_addr);
01015         if ((index <= 3) && !b43_new_kidx_api(dev)) {

01348 static void handle_irq_noise(struct b43_wldev *dev)
01349 {
01350         struct b43_phy_g *phy = dev->phy.g;
01351         u16 tmp;
01352         u8 noise[4];
01353         u8 i, j;
01354         s32 average;
01355 
01356         /* Bottom half of Link Quality calculation. */
01357 
01358         if (dev->phy.type != B43_PHYTYPE_G)
01359                 return;
01360 
01361         /* Possible race condition: It might be possible that the user
01362          * changed to a different channel in the meantime since we
01363          * started the calculation. We ignore that fact, since it's
01364          * not really that much of a problem. The background noise is
01365          * an estimation only anyway. Slightly wrong results will get damped
01366          * by the averaging of the 8 sample rounds. Additionally the
01367          * value is shortlived. So it will be replaced by the next noise
01368          * calculation round soon. */
01369 
01370         B43_WARN_ON(!dev->noisecalc.calculation_running);
01371         *((__le32 *)noise) = cpu_to_le32(b43_jssi_read(dev));
01372         if (noise[0] == 0x7F || noise[1] == 0x7F ||
01373             noise[2] == 0x7F || noise[3] == 0x7F)
01374                 goto generate_new;
01375 
01376         /* Get the noise samples. */
01377         B43_WARN_ON(dev->noisecalc.nr_samples >= 8);
01378         i = dev->noisecalc.nr_samples;
01379         noise[0] = clamp_val(noise[0], 0, ARRAY_SIZE(phy->nrssi_lt) - 1);
01380         noise[1] = clamp_val(noise[1], 0, ARRAY_SIZE(phy->nrssi_lt) - 1);
01381         noise[2] = clamp_val(noise[2], 0, ARRAY_SIZE(phy->nrssi_lt) - 1);
01382         noise[3] = clamp_val(noise[3], 0, ARRAY_SIZE(phy->nrssi_lt) - 1);
01383         dev->noisecalc.samples[i][0] = phy->nrssi_lt[noise[0]];
01384         dev->noisecalc.samples[i][1] = phy->nrssi_lt[noise[1]];
01385         dev->noisecalc.samples[i][2] = phy->nrssi_lt[noise[2]];
01386         dev->noisecalc.samples[i][3] = phy->nrssi_lt[noise[3]];
01387         dev->noisecalc.nr_samples++;
01388         if (dev->noisecalc.nr_samples == 8) {
01389                 /* Calculate the Link Quality by the noise samples. */
01390                 average = 0;
01391                 for (i = 0; i < 8; i++) {
01392                         for (j = 0; j < 4; j++)
01393                                 average += dev->noisecalc.samples[i][j];
01394                 }
01395                 average /= (8 * 4);
01396                 average *= 125;
01397                 average += 64;
01398                 average /= 128;
01399                 tmp = b43_shm_read16(dev, B43_SHM_SHARED, 0x40C);
01400                 tmp = (tmp / 128) & 0x1F;
01401                 if (tmp >= 8)
01402                         average += 2;
01403                 else
01404                         average -= 25;
01405                 if (tmp == 8)
01406                         average -= 72;
01407                 else
01408                         average -= 48;
01409 
01410                 dev->stats.link_noise = average;
01411                 dev->noisecalc.calculation_running = 0;
01412                 return;
01413         }
01414 generate_new:
01415         b43_generate_noise_sample(dev);
01416 }

01455 static void b43_write_template_common(struct b43_wldev *dev,
01456                                       const u8 *data, u16 size,
01457                                       u16 ram_offset,
01458                                       u16 shm_size_offset, u8 rate)
01459 {
01460         u32 i, tmp;
01461         struct b43_plcp_hdr4 plcp;
01462

01465         b43_ram_write(dev, ram_offset, le32_to_cpu(plcp.data));
01466         ram_offset += sizeof(u32);
01467         /* The PLCP is 6 bytes long, but we only wrote 4 bytes, yet.
01468          * So leave the first two bytes of the next write blank.
01469          */
01470         tmp = (u32) (data[0]) << 16;
01471         tmp |= (u32) (data[1]) << 24;
01472         b43_ram_write(dev, ram_offset, tmp);

01475                 tmp = (u32) (data[i + 0]);
01476                 if (i + 1 < size)
01477                         tmp |= (u32) (data[i + 1]) << 8;
01478                 if (i + 2 < size)
01479                         tmp |= (u32) (data[i + 2]) << 16;
01480                 if (i + 3 < size)
01481                         tmp |= (u32) (data[i + 3]) << 24;
01482                 b43_ram_write(dev, ram_offset + i - 2, tmp);
01483         }
01484         b43_shm_write16(dev, B43_SHM_SHARED, shm_size_offset,
01485                         size + sizeof(struct b43_plcp_hdr6));

01559         ctl = b43_shm_read16(dev, B43_SHM_SHARED, B43_SHM_SH_BEACPHYCTL);

01566                 ctl |= B43_TXH_PHY_ENC_CCK;

01570

01572          * and write them to SHM. */

01574         variable_len = len - offsetof(struct ieee80211_mgmt, u.beacon.variable);
01575         for (i = 0; i < variable_len - 2; ) {
01576                 uint8_t ie_id, ie_len;
01577 
01578                 ie_id = ie[i];
01579                 ie_len = ie[i + 1];
01580                 if (ie_id == 5) {
01581                         u16 tim_position;
01582                         u16 dtim_period;
01583                         /* This is the TIM Information Element */
01584 
01585                         /* Check whether the ie_len is in the beacon data range. */
01586                         if (variable_len < ie_len + 2 + i)
01587                                 break;
01588                         /* A valid TIM is at least 4 bytes long. */
01589                         if (ie_len < 4)
01590                                 break;

01592 
01593                         tim_position = sizeof(struct b43_plcp_hdr6);
01594                         tim_position += offsetof(struct ieee80211_mgmt, u.beacon.variable);
01595                         tim_position += i;
01596 
01597                         dtim_period = ie[i + 3];
01598 
01599                         b43_shm_write16(dev, B43_SHM_SHARED,
01600                                         B43_SHM_SH_TIMBPOS, tim_position);
01601                         b43_shm_write16(dev, B43_SHM_SHARED,
01602                                         B43_SHM_SH_DTIMPER, dtim_period);
01603                         break;
01604                 }
01605                 i += ie_len + 2;
01606         }
01607         if (!tim_found) {
01608                 /*
01609                  * If ucode wants to modify TIM do it behind the beacon, this
01610                  * will happen, for example, when doing mesh networking.
01611                  */
01612                 b43_shm_write16(dev, B43_SHM_SHARED,
01613                                 B43_SHM_SH_TIMBPOS,
01614                                 len + sizeof(struct b43_plcp_hdr6));
01615                 b43_shm_write16(dev, B43_SHM_SHARED,
01616                                 B43_SHM_SH_DTIMPER, 0);
01617         }
01618         b43dbg(dev->wl, "Updated beacon template at 0x%x\n", ram_offset);
01619 }

01641 static void handle_irq_beacon(struct b43_wldev *dev)
01642 {
01643         struct b43_wl *wl = dev->wl;
01644         u32 cmd, beacon0_valid, beacon1_valid;
01645 
01646         if (!b43_is_mode(wl, NL80211_IFTYPE_AP) &&
01647             !b43_is_mode(wl, NL80211_IFTYPE_MESH_POINT) &&
01648             !b43_is_mode(wl, NL80211_IFTYPE_ADHOC))
01649                 return;
01650 
01651         /* This is the bottom half of the asynchronous beacon update. */
01652 
01653         /* Ignore interrupt in the future. */
01654         dev->irq_mask &= ~B43_IRQ_BEACON;
01655 
01656         cmd = b43_read32(dev, B43_MMIO_MACCMD);
01657         beacon0_valid = (cmd & B43_MACCMD_BEACON0_VALID);
01658         beacon1_valid = (cmd & B43_MACCMD_BEACON1_VALID);
01659 
01660         /* Schedule interrupt manually, if busy. */
01661         if (beacon0_valid && beacon1_valid) {
01662                 b43_write32(dev, B43_MMIO_GEN_IRQ_REASON, B43_IRQ_BEACON);
01663                 dev->irq_mask |= B43_IRQ_BEACON;
01664                 return;
01665         }
01666 
01667         if (unlikely(wl->beacon_templates_virgin)) {
01668                 /* We never uploaded a beacon before.
01669                  * Upload both templates now, but only mark one valid. */
01670                 wl->beacon_templates_virgin = 0;
01671                 b43_upload_beacon0(dev);
01672                 b43_upload_beacon1(dev);
01673                 cmd = b43_read32(dev, B43_MMIO_MACCMD);
01674                 cmd |= B43_MACCMD_BEACON0_VALID;
01675                 b43_write32(dev, B43_MMIO_MACCMD, cmd);
01676         } else {
01677                 if (!beacon0_valid) {
01678                         b43_upload_beacon0(dev);
01679                         cmd = b43_read32(dev, B43_MMIO_MACCMD);
01680                         cmd |= B43_MACCMD_BEACON0_VALID;
01681                         b43_write32(dev, B43_MMIO_MACCMD, cmd);
01682                 } else if (!beacon1_valid) {
01683                         b43_upload_beacon1(dev);
01684                         cmd = b43_read32(dev, B43_MMIO_MACCMD);
01685                         cmd |= B43_MACCMD_BEACON1_VALID;
01686                         b43_write32(dev, B43_MMIO_MACCMD, cmd);
01687                 }
01688         }
01689 }

01735 /* Asynchronously update the packet templates in template RAM.

01737 static void b43_update_templates(struct b43_wl *wl)
01738 {
01739         struct sk_buff *beacon;
01740 
01741         /* This is the top half of the ansynchronous beacon update.
01742          * The bottom half is the beacon IRQ.
01743          * Beacon update must be asynchronous to avoid sending an
01744          * invalid beacon. This can happen for example, if the firmware
01745          * transmits a beacon while we are updating it. */
01746 
01747         /* We could modify the existing beacon and set the aid bit in
01748          * the TIM field, but that would probably require resizing and
01749          * moving of data within the beacon template.

01751         beacon = ieee80211_beacon_get(wl->hw, wl->vif);
01752         if (unlikely(!beacon))

01756                 dev_kfree_skb_any(wl->current_beacon);

01758         wl->beacon0_uploaded = 0;
01759         wl->beacon1_uploaded = 0;
01760         ieee80211_queue_work(wl->hw, &wl->beacon_update_trigger);

01873 static void b43_do_interrupt_thread(struct b43_wldev *dev)
01874 {
01875         u32 reason;
01876         u32 dma_reason[ARRAY_SIZE(dev->dma_reason)];
01877         u32 merged_dma_reason = 0;
01878         int i;
01879 
01880         if (unlikely(b43_status(dev) != B43_STAT_STARTED))
01881                 return;
01882 
01883         reason = dev->irq_reason;
01884         for (i = 0; i < ARRAY_SIZE(dma_reason); i++) {
01885                 dma_reason[i] = dev->dma_reason[i];
01886                 merged_dma_reason |= dma_reason[i];
01887         }
01888 
01889         if (unlikely(reason & B43_IRQ_MAC_TXERR))
01890                 b43err(dev->wl, "MAC transmission error\n");
01891 
01892         if (unlikely(reason & B43_IRQ_PHY_TXERR)) {
01893                 b43err(dev->wl, "PHY transmission error\n");
01894                 rmb();
01895                 if (unlikely(atomic_dec_and_test(&dev->phy.txerr_cnt))) {
01896                         atomic_set(&dev->phy.txerr_cnt,
01897                                    B43_PHY_TX_BADNESS_LIMIT);
01898                         b43err(dev->wl, "Too many PHY TX errors, "
01899                                         "restarting the controller\n");
01900                         b43_controller_restart(dev, "PHY TX errors");
01901                 }
01902         }
01903 
01904         if (unlikely(merged_dma_reason & (B43_DMAIRQ_FATALMASK |
01905                                           B43_DMAIRQ_NONFATALMASK))) {
01906                 if (merged_dma_reason & B43_DMAIRQ_FATALMASK) {
01907                         b43err(dev->wl, "Fatal DMA error: "
01908                                "0x%08X, 0x%08X, 0x%08X, "
01909                                "0x%08X, 0x%08X, 0x%08X\n",
01910                                dma_reason[0], dma_reason[1],
01911                                dma_reason[2], dma_reason[3],
01912                                dma_reason[4], dma_reason[5]);
01913                         b43err(dev->wl, "This device does not support DMA "
01914                                "on your system. It will now be switched to PIO.\n");
01915                         /* Fall back to PIO transfers if we get fatal DMA errors! */
01916                         dev->use_pio = 1;
01917                         b43_controller_restart(dev, "DMA error");
01918                         return;
01919                 }
01920                 if (merged_dma_reason & B43_DMAIRQ_NONFATALMASK) {
01921                         b43err(dev->wl, "DMA error: "
01922                                "0x%08X, 0x%08X, 0x%08X, "
01923                                "0x%08X, 0x%08X, 0x%08X\n",
01924                                dma_reason[0], dma_reason[1],
01925                                dma_reason[2], dma_reason[3],
01926                                dma_reason[4], dma_reason[5]);
01927                 }
01928         }
01929 
01930         if (unlikely(reason & B43_IRQ_UCODE_DEBUG))
01931                 handle_irq_ucode_debug(dev);
01932         if (reason & B43_IRQ_TBTT_INDI)
01933                 handle_irq_tbtt_indication(dev);
01934         if (reason & B43_IRQ_ATIM_END)
01935                 handle_irq_atim_end(dev);
01936         if (reason & B43_IRQ_BEACON)
01937                 handle_irq_beacon(dev);
01938         if (reason & B43_IRQ_PMQ)
01939                 handle_irq_pmq(dev);
01940         if (reason & B43_IRQ_TXFIFO_FLUSH_OK)
01941                 ;/* TODO */
01942         if (reason & B43_IRQ_NOISESAMPLE_OK)
01943                 handle_irq_noise(dev);
01944 
01945         /* Check the DMA reason registers for received data. */
01946         if (dma_reason[0] & B43_DMAIRQ_RX_DONE) {
01947                 if (b43_using_pio_transfers(dev))
01948                         b43_pio_rx(dev->pio.rx_queue);
01949                 else
01950                         b43_dma_rx(dev->dma.rx_ring);
01951         }
01952         B43_WARN_ON(dma_reason[1] & B43_DMAIRQ_RX_DONE);
01953         B43_WARN_ON(dma_reason[2] & B43_DMAIRQ_RX_DONE);
01954         B43_WARN_ON(dma_reason[3] & B43_DMAIRQ_RX_DONE);
01955         B43_WARN_ON(dma_reason[4] & B43_DMAIRQ_RX_DONE);
01956         B43_WARN_ON(dma_reason[5] & B43_DMAIRQ_RX_DONE);
01957 
01958         if (reason & B43_IRQ_TX_OK)
01959                 handle_irq_transmit_status(dev);
01960 
01961         /* Re-enable interrupts on the device by restoring the current interrupt mask. */
01962         b43_write32(dev, B43_MMIO_GEN_IRQ_MASK, dev->irq_mask);
01963 
01964 #if B43_DEBUG
01965         if (b43_debug(dev, B43_DBG_VERBOSESTATS)) {
01966                 dev->irq_count++;
01967                 for (i = 0; i < ARRAY_SIZE(dev->irq_bit_count); i++) {
01968                         if (reason & (1 << i))
01969                                 dev->irq_bit_count[i]++;
01970                 }
01971         }
01972 #endif
01973 }

01988 static irqreturn_t b43_do_interrupt(struct b43_wldev *dev)
01989 {
01990         u32 reason;
01991 
01992         /* This code runs under wl->hardirq_lock, but _only_ on non-SDIO busses.
01993          * On SDIO, this runs under wl->mutex. */
01994 
01995         reason = b43_read32(dev, B43_MMIO_GEN_IRQ_REASON);
01996         if (reason == 0xffffffff)       /* shared IRQ */
01997                 return IRQ_NONE;
01998         reason &= dev->irq_mask;
01999         if (!reason)
02000                 return IRQ_NONE;
02001 
02002         dev->dma_reason[0] = b43_read32(dev, B43_MMIO_DMA0_REASON)
02003             & 0x0001DC00;
02004         dev->dma_reason[1] = b43_read32(dev, B43_MMIO_DMA1_REASON)
02005             & 0x0000DC00;
02006         dev->dma_reason[2] = b43_read32(dev, B43_MMIO_DMA2_REASON)
02007             & 0x0000DC00;
02008         dev->dma_reason[3] = b43_read32(dev, B43_MMIO_DMA3_REASON)
02009             & 0x0001DC00;
02010         dev->dma_reason[4] = b43_read32(dev, B43_MMIO_DMA4_REASON)
02011             & 0x0000DC00;
02012 /* Unused ring
02013         dev->dma_reason[5] = b43_read32(dev, B43_MMIO_DMA5_REASON)
02014             & 0x0000DC00;
02015 */
02016 
02017         /* ACK the interrupt. */
02018         b43_write32(dev, B43_MMIO_GEN_IRQ_REASON, reason);
02019         b43_write32(dev, B43_MMIO_DMA0_REASON, dev->dma_reason[0]);
02020         b43_write32(dev, B43_MMIO_DMA1_REASON, dev->dma_reason[1]);
02021         b43_write32(dev, B43_MMIO_DMA2_REASON, dev->dma_reason[2]);
02022         b43_write32(dev, B43_MMIO_DMA3_REASON, dev->dma_reason[3]);
02023         b43_write32(dev, B43_MMIO_DMA4_REASON, dev->dma_reason[4]);
02024 /* Unused ring
02025         b43_write32(dev, B43_MMIO_DMA5_REASON, dev->dma_reason[5]);
02026 */
02027 
02028         /* Disable IRQs on the device. The IRQ thread handler will re-enable them. */
02029         b43_write32(dev, B43_MMIO_GEN_IRQ_MASK, 0);
02030         /* Save the reason bitmasks for the IRQ thread handler. */
02031         dev->irq_reason = reason;
02032 
02033         return IRQ_WAKE_THREAD;
02034 }

02097 int b43_do_request_fw(struct b43_request_fw_context *ctx,
02098                       const char *name,
02099                       struct b43_firmware_file *fw)
02100 {
02101         const struct firmware *blob;
02102         struct b43_fw_header *hdr;
02103         u32 size;
02104         int err;
02105 
02106         if (!name) {
02107                 /* Don't fetch anything. Free possibly cached firmware. */
02108                 /* FIXME: We should probably keep it anyway, to save some headache
02109                  * on suspend/resume with multiband devices. */
02110                 b43_do_release_fw(fw);
02111                 return 0;
02112         }
02113         if (fw->filename) {
02114                 if ((fw->type == ctx->req_type) &&
02115                     (strcmp(fw->filename, name) == 0))
02116                         return 0; /* Already have this fw. */
02117                 /* Free the cached firmware first. */
02118                 /* FIXME: We should probably do this later after we successfully
02119                  * got the new fw. This could reduce headache with multiband devices.
02120                  * We could also redesign this to cache the firmware for all possible
02121                  * bands all the time. */
02122                 b43_do_release_fw(fw);
02123         }
02124 
02125         switch (ctx->req_type) {
02126         case B43_FWTYPE_PROPRIETARY:
02127                 snprintf(ctx->fwname, sizeof(ctx->fwname),
02128                          "b43%s/%s.fw",
02129                          modparam_fwpostfix, name);
02130                 break;
02131         case B43_FWTYPE_OPENSOURCE:
02132                 snprintf(ctx->fwname, sizeof(ctx->fwname),
02133                          "b43-open%s/%s.fw",
02134                          modparam_fwpostfix, name);
02135                 break;
02136         default:
02137                 B43_WARN_ON(1);
02138                 return -ENOSYS;
02139         }
02140         err = request_firmware(&blob, ctx->fwname, ctx->dev->dev->dev);
02141         if (err == -ENOENT) {
02142                 snprintf(ctx->errors[ctx->req_type],
02143                          sizeof(ctx->errors[ctx->req_type]),
02144                          "Firmware file \"%s\" not found\n", ctx->fwname);
02145                 return err;
02146         } else if (err) {
02147                 snprintf(ctx->errors[ctx->req_type],
02148                          sizeof(ctx->errors[ctx->req_type]),
02149                          "Firmware file \"%s\" request failed (err=%d)\n",
02150                          ctx->fwname, err);
02151                 return err;
02152         }
02153         if (blob->size < sizeof(struct b43_fw_header))
02154                 goto err_format;
02155         hdr = (struct b43_fw_header *)(blob->data);
02156         switch (hdr->type) {
02157         case B43_FW_TYPE_UCODE:
02158         case B43_FW_TYPE_PCM:
02159                 size = be32_to_cpu(hdr->size);
02160                 if (size != blob->size - sizeof(struct b43_fw_header))
02161                         goto err_format;
02162                 /* fallthrough */
02163         case B43_FW_TYPE_IV:
02164                 if (hdr->ver != 1)
02165                         goto err_format;
02166                 break;
02167         default:
02168                 goto err_format;
02169         }
02170 
02171         fw->data = blob;
02172         fw->filename = name;
02173         fw->type = ctx->req_type;
02174 
02175         return 0;
02176 
02177 err_format:
02178         snprintf(ctx->errors[ctx->req_type],
02179                  sizeof(ctx->errors[ctx->req_type]),
02180                  "Firmware file \"%s\" format error.\n", ctx->fwname);
02181         release_firmware(blob);
02182 
02183         return -EPROTO;
02184 }

02399 static int b43_request_firmware(struct b43_wldev *dev)
02400 {
02401         struct b43_request_fw_context *ctx;
02402         unsigned int i;
02403         int err;
02404         const char *errmsg;
02405 
02406         ctx = kzalloc(sizeof(*ctx), GFP_KERNEL);
02407         if (!ctx)
02408                 return -ENOMEM;
02409         ctx->dev = dev;
02410 
02411         ctx->req_type = B43_FWTYPE_PROPRIETARY;
02412         err = b43_try_request_fw(ctx);
02413         if (!err)
02414                 goto out; /* Successfully loaded it. */
02415         err = ctx->fatal_failure;
02416         if (err)
02417                 goto out;
02418 
02419         ctx->req_type = B43_FWTYPE_OPENSOURCE;
02420         err = b43_try_request_fw(ctx);
02421         if (!err)
02422                 goto out; /* Successfully loaded it. */
02423         err = ctx->fatal_failure;
02424         if (err)
02425                 goto out;
02426 
02427         /* Could not find a usable firmware. Print the errors. */
02428         for (i = 0; i < B43_NR_FWTYPES; i++) {
02429                 errmsg = ctx->errors[i];
02430                 if (strlen(errmsg))
02431                         b43err(dev->wl, errmsg);
02432         }
02433         b43_print_fw_helptext(dev->wl, 1);
02434         err = -ENOENT;
02435 
02436 out:
02437         kfree(ctx);
02438         return err;
02439 }

02441 static int b43_upload_microcode(struct b43_wldev *dev)
02442 {
02443         struct wiphy *wiphy = dev->wl->hw->wiphy;
02444         const size_t hdr_len = sizeof(struct b43_fw_header);
02445         const __be32 *data;
02446         unsigned int i, len;
02447         u16 fwrev, fwpatch, fwdate, fwtime;
02448         u32 tmp, macctl;
02449         int err = 0;
02450 
02451         /* Jump the microcode PSM to offset 0 */
02452         macctl = b43_read32(dev, B43_MMIO_MACCTL);
02453         B43_WARN_ON(macctl & B43_MACCTL_PSM_RUN);
02454         macctl |= B43_MACCTL_PSM_JMP0;
02455         b43_write32(dev, B43_MMIO_MACCTL, macctl);
02456         /* Zero out all microcode PSM registers and shared memory. */
02457         for (i = 0; i < 64; i++)
02458                 b43_shm_write16(dev, B43_SHM_SCRATCH, i, 0);
02459         for (i = 0; i < 4096; i += 2)
02460                 b43_shm_write16(dev, B43_SHM_SHARED, i, 0);
02461 
02462         /* Upload Microcode. */
02463         data = (__be32 *) (dev->fw.ucode.data->data + hdr_len);
02464         len = (dev->fw.ucode.data->size - hdr_len) / sizeof(__be32);
02465         b43_shm_control_word(dev, B43_SHM_UCODE | B43_SHM_AUTOINC_W, 0x0000);
02466         for (i = 0; i < len; i++) {
02467                 b43_write32(dev, B43_MMIO_SHM_DATA, be32_to_cpu(data[i]));
02468                 udelay(10);
02469         }
02470 
02471         if (dev->fw.pcm.data) {
02472                 /* Upload PCM data. */
02473                 data = (__be32 *) (dev->fw.pcm.data->data + hdr_len);
02474                 len = (dev->fw.pcm.data->size - hdr_len) / sizeof(__be32);
02475                 b43_shm_control_word(dev, B43_SHM_HW, 0x01EA);
02476                 b43_write32(dev, B43_MMIO_SHM_DATA, 0x00004000);
02477                 /* No need for autoinc bit in SHM_HW */
02478                 b43_shm_control_word(dev, B43_SHM_HW, 0x01EB);
02479                 for (i = 0; i < len; i++) {
02480                         b43_write32(dev, B43_MMIO_SHM_DATA, be32_to_cpu(data[i]));
02481                         udelay(10);
02482                 }
02483         }
02484 
02485         b43_write32(dev, B43_MMIO_GEN_IRQ_REASON, B43_IRQ_ALL);
02486 
02487         /* Start the microcode PSM */
02488         macctl = b43_read32(dev, B43_MMIO_MACCTL);
02489         macctl &= ~B43_MACCTL_PSM_JMP0;
02490         macctl |= B43_MACCTL_PSM_RUN;
02491         b43_write32(dev, B43_MMIO_MACCTL, macctl);
02492 
02493         /* Wait for the microcode to load and respond */
02494         i = 0;
02495         while (1) {
02496                 tmp = b43_read32(dev, B43_MMIO_GEN_IRQ_REASON);
02497                 if (tmp == B43_IRQ_MAC_SUSPENDED)
02498                         break;
02499                 i++;
02500                 if (i >= 20) {
02501                         b43err(dev->wl, "Microcode not responding\n");
02502                         b43_print_fw_helptext(dev->wl, 1);
02503                         err = -ENODEV;
02504                         goto error;
02505                 }
02506                 msleep(50);
02507         }
02508         b43_read32(dev, B43_MMIO_GEN_IRQ_REASON);       /* dummy read */
02509 
02510         /* Get and check the revisions. */
02511         fwrev = b43_shm_read16(dev, B43_SHM_SHARED, B43_SHM_SH_UCODEREV);
02512         fwpatch = b43_shm_read16(dev, B43_SHM_SHARED, B43_SHM_SH_UCODEPATCH);
02513         fwdate = b43_shm_read16(dev, B43_SHM_SHARED, B43_SHM_SH_UCODEDATE);
02514         fwtime = b43_shm_read16(dev, B43_SHM_SHARED, B43_SHM_SH_UCODETIME);
02515 
02516         if (fwrev <= 0x128) {
02517                 b43err(dev->wl, "YOUR FIRMWARE IS TOO OLD. Firmware from "
02518                        "binary drivers older than version 4.x is unsupported. "
02519                        "You must upgrade your firmware files.\n");
02520                 b43_print_fw_helptext(dev->wl, 1);
02521                 err = -EOPNOTSUPP;
02522                 goto error;
02523         }
02524         dev->fw.rev = fwrev;
02525         dev->fw.patch = fwpatch;
02526         if (dev->fw.rev >= 598)
02527                 dev->fw.hdr_format = B43_FW_HDR_598;
02528         else if (dev->fw.rev >= 410)
02529                 dev->fw.hdr_format = B43_FW_HDR_410;
02530         else
02531                 dev->fw.hdr_format = B43_FW_HDR_351;
02532         dev->fw.opensource = (fwdate == 0xFFFF);
02533 
02534         /* Default to use-all-queues. */
02535         dev->wl->hw->queues = dev->wl->mac80211_initially_registered_queues;
02536         dev->qos_enabled = !!modparam_qos;
02537         /* Default to firmware/hardware crypto acceleration. */
02538         dev->hwcrypto_enabled = 1;
02539 
02540         if (dev->fw.opensource) {
02541                 u16 fwcapa;
02542 
02543                 /* Patchlevel info is encoded in the "time" field. */
02544                 dev->fw.patch = fwtime;
02545                 b43info(dev->wl, "Loading OpenSource firmware version %u.%u\n",
02546                         dev->fw.rev, dev->fw.patch);
02547 
02548                 fwcapa = b43_fwcapa_read(dev);
02549                 if (!(fwcapa & B43_FWCAPA_HWCRYPTO) || dev->fw.pcm_request_failed) {
02550                         b43info(dev->wl, "Hardware crypto acceleration not supported by firmware\n");
02551                         /* Disable hardware crypto and fall back to software crypto. */
02552                         dev->hwcrypto_enabled = 0;
02553                 }
02554                 if (!(fwcapa & B43_FWCAPA_QOS)) {
02555                         b43info(dev->wl, "QoS not supported by firmware\n");
02556                         /* Disable QoS. Tweak hw->queues to 1. It will be restored before
02557                          * ieee80211_unregister to make sure the networking core can
02558                          * properly free possible resources. */
02559                         dev->wl->hw->queues = 1;
02560                         dev->qos_enabled = 0;
02561                 }
02562         } else {
02563                 b43info(dev->wl, "Loading firmware version %u.%u "
02564                         "(20%.2i-%.2i-%.2i %.2i:%.2i:%.2i)\n",
02565                         fwrev, fwpatch,
02566                         (fwdate >> 12) & 0xF, (fwdate >> 8) & 0xF, fwdate & 0xFF,
02567                         (fwtime >> 11) & 0x1F, (fwtime >> 5) & 0x3F, fwtime & 0x1F);
02568                 if (dev->fw.pcm_request_failed) {
02569                         b43warn(dev->wl, "No \"pcm5.fw\" firmware file found. "
02570                                 "Hardware accelerated cryptography is disabled.\n");
02571                         b43_print_fw_helptext(dev->wl, 0);
02572                 }
02573         }
02574 
02575         snprintf(wiphy->fw_version, sizeof(wiphy->fw_version), "%u.%u",
02576                         dev->fw.rev, dev->fw.patch);
02577         wiphy->hw_version = dev->dev->core_id;
02578 
02579         if (dev->fw.hdr_format == B43_FW_HDR_351) {
02580                 /* We're over the deadline, but we keep support for old fw
02581                  * until it turns out to be in major conflict with something new. */
02582                 b43warn(dev->wl, "You are using an old firmware image. "
02583                         "Support for old firmware will be removed soon "
02584                         "(official deadline was July 2008).\n");
02585                 b43_print_fw_helptext(dev->wl, 0);
02586         }
02587 
02588         return 0;
02589 
02590 error:
02591         macctl = b43_read32(dev, B43_MMIO_MACCTL);
02592         macctl &= ~B43_MACCTL_PSM_RUN;
02593         macctl |= B43_MACCTL_PSM_JMP0;
02594         b43_write32(dev, B43_MMIO_MACCTL, macctl);
02595 
02596         return err;
02597 }

02599 static int b43_write_initvals(struct b43_wldev *dev,
02600                               const struct b43_iv *ivals,
02601                               size_t count,
02602                               size_t array_size)
02603 {
02604         const struct b43_iv *iv;
02605         u16 offset;
02606         size_t i;
02607         bool bit32;
02608 
02609         BUILD_BUG_ON(sizeof(struct b43_iv) != 6);
02610         iv = ivals;
02611         for (i = 0; i < count; i++) {
02612                 if (array_size < sizeof(iv->offset_size))
02613                         goto err_format;
02614                 array_size -= sizeof(iv->offset_size);
02615                 offset = be16_to_cpu(iv->offset_size);
02616                 bit32 = !!(offset & B43_IV_32BIT);
02617                 offset &= B43_IV_OFFSET_MASK;
02618                 if (offset >= 0x1000)
02619                         goto err_format;
02620                 if (bit32) {
02621                         u32 value;
02622 
02623                         if (array_size < sizeof(iv->data.d32))
02624                                 goto err_format;
02625                         array_size -= sizeof(iv->data.d32);
02626 
02627                         value = get_unaligned_be32(&iv->data.d32);
02628                         b43_write32(dev, offset, value);
02629 
02630                         iv = (const struct b43_iv *)((const uint8_t *)iv +
02631                                                         sizeof(__be16) +
02632                                                         sizeof(__be32));
02633                 } else {
02634                         u16 value;
02635 
02636                         if (array_size < sizeof(iv->data.d16))
02637                                 goto err_format;
02638                         array_size -= sizeof(iv->data.d16);
02639 
02640                         value = be16_to_cpu(iv->data.d16);
02641                         b43_write16(dev, offset, value);
02642 
02643                         iv = (const struct b43_iv *)((const uint8_t *)iv +
02644                                                         sizeof(__be16) +
02645                                                         sizeof(__be16));
02646                 }
02647         }
02648         if (array_size)
02649                 goto err_format;
02650 
02651         return 0;
02652 
02653 err_format:
02654         b43err(dev->wl, "Initial Values Firmware file-format error.\n");
02655         b43_print_fw_helptext(dev->wl, 1);
02656 
02657         return -EPROTO;
02658 }

02814 void b43_mac_suspend(struct b43_wldev *dev)
02815 {
02816         int i;
02817         u32 tmp;
02818

02825                             b43_read32(dev, B43_MMIO_MACCTL)

02828                 b43_read32(dev, B43_MMIO_MACCTL);

02838                         if (tmp & B43_IRQ_MAC_SUSPENDED)
02839                                 goto out;

02877 static void b43_adjust_opmode(struct b43_wldev *dev)
02878 {
02879         struct b43_wl *wl = dev->wl;
02880         u32 ctl;
02881         u16 cfp_pretbtt;
02882 
02883         ctl = b43_read32(dev, B43_MMIO_MACCTL);
02884         /* Reset status to STA infrastructure mode. */
02885         ctl &= ~B43_MACCTL_AP;
02886         ctl &= ~B43_MACCTL_KEEP_CTL;
02887         ctl &= ~B43_MACCTL_KEEP_BADPLCP;
02888         ctl &= ~B43_MACCTL_KEEP_BAD;
02889         ctl &= ~B43_MACCTL_PROMISC;
02890         ctl &= ~B43_MACCTL_BEACPROMISC;
02891         ctl |= B43_MACCTL_INFRA;
02892 
02893         if (b43_is_mode(wl, NL80211_IFTYPE_AP) ||
02894             b43_is_mode(wl, NL80211_IFTYPE_MESH_POINT))
02895                 ctl |= B43_MACCTL_AP;
02896         else if (b43_is_mode(wl, NL80211_IFTYPE_ADHOC))
02897                 ctl &= ~B43_MACCTL_INFRA;
02898 
02899         if (wl->filter_flags & FIF_CONTROL)
02900                 ctl |= B43_MACCTL_KEEP_CTL;
02901         if (wl->filter_flags & FIF_FCSFAIL)
02902                 ctl |= B43_MACCTL_KEEP_BAD;
02903         if (wl->filter_flags & FIF_PLCPFAIL)
02904                 ctl |= B43_MACCTL_KEEP_BADPLCP;
02905         if (wl->filter_flags & FIF_PROMISC_IN_BSS)
02906                 ctl |= B43_MACCTL_PROMISC;
02907         if (wl->filter_flags & FIF_BCN_PRBRESP_PROMISC)
02908                 ctl |= B43_MACCTL_BEACPROMISC;
02909 
02910         /* Workaround: On old hardware the HW-MAC-address-filter
02911          * doesn't work properly, so always run promisc in filter
02912          * it in software. */
02913         if (dev->dev->core_rev <= 4)
02914                 ctl |= B43_MACCTL_PROMISC;
02915 
02916         b43_write32(dev, B43_MMIO_MACCTL, ctl);
02917 
02918         cfp_pretbtt = 2;
02919         if ((ctl & B43_MACCTL_INFRA) && !(ctl & B43_MACCTL_AP)) {
02920                 if (dev->dev->chip_id == 0x4306 &&
02921                     dev->dev->chip_rev == 3)
02922                         cfp_pretbtt = 100;
02923                 else
02924                         cfp_pretbtt = 50;
02925         }
02926         b43_write16(dev, 0x612, cfp_pretbtt);
02927 
02928         /* FIXME: We don't currently implement the PMQ mechanism,
02929          *        so always disable it. If we want to implement PMQ,
02930          *        we need to enable it here (clear DISCPMQ) in AP mode.
02931          */
02932         if (0  /* ctl & B43_MACCTL_AP */) {
02933                 b43_write32(dev, B43_MMIO_MACCTL,
02934                             b43_read32(dev, B43_MMIO_MACCTL)
02935                             & ~B43_MACCTL_DISCPMQ);
02936         } else {
02937                 b43_write32(dev, B43_MMIO_MACCTL,
02938                             b43_read32(dev, B43_MMIO_MACCTL)
02939                             | B43_MACCTL_DISCPMQ);
02940         }
02941 }

03029  * http://bcm-specs.sipsolutions.net/ChipInit
03030  */
03031 static int b43_chip_init(struct b43_wldev *dev)
03032 {
03033         struct b43_phy *phy = &dev->phy;
03034         int err;
03035         u32 macctl;
03036         u16 value16;
03037 
03038         /* Initialize the MAC control */
03039         macctl = B43_MACCTL_IHR_ENABLED | B43_MACCTL_SHM_ENABLED;
03040         if (dev->phy.gmode)
03041                 macctl |= B43_MACCTL_GMODE;
03042         macctl |= B43_MACCTL_INFRA;
03043         b43_write32(dev, B43_MMIO_MACCTL, macctl);
03044 
03045         err = b43_request_firmware(dev);
03046         if (err)
03047                 goto out;
03048         err = b43_upload_microcode(dev);
03049         if (err)
03050                 goto out;       /* firmware is released later */
03051 
03052         err = b43_gpio_init(dev);
03053         if (err)
03054                 goto out;       /* firmware is released later */
03055 
03056         err = b43_upload_initvals(dev);
03057         if (err)
03058                 goto err_gpio_clean;
03059 
03060         /* Turn the Analog on and initialize the PHY. */
03061         phy->ops->switch_analog(dev, 1);
03062         err = b43_phy_init(dev);
03063         if (err)
03064                 goto err_gpio_clean;
03065 
03066         /* Disable Interference Mitigation. */
03067         if (phy->ops->interf_mitigation)
03068                 phy->ops->interf_mitigation(dev, B43_INTERFMODE_NONE);
03069 
03070         /* Select the antennae */
03071         if (phy->ops->set_rx_antenna)
03072                 phy->ops->set_rx_antenna(dev, B43_ANTENNA_DEFAULT);
03073         b43_mgmtframe_txantenna(dev, B43_ANTENNA_DEFAULT);
03074 
03075         if (phy->type == B43_PHYTYPE_B) {
03076                 value16 = b43_read16(dev, 0x005E);
03077                 value16 |= 0x0004;
03078                 b43_write16(dev, 0x005E, value16);
03079         }
03080         b43_write32(dev, 0x0100, 0x01000000);
03081         if (dev->dev->core_rev < 5)
03082                 b43_write32(dev, 0x010C, 0x01000000);
03083 
03084         b43_write32(dev, B43_MMIO_MACCTL, b43_read32(dev, B43_MMIO_MACCTL)
03085                     & ~B43_MACCTL_INFRA);
03086         b43_write32(dev, B43_MMIO_MACCTL, b43_read32(dev, B43_MMIO_MACCTL)
03087                     | B43_MACCTL_INFRA);
03088 
03089         /* Probe Response Timeout value */

03091         b43_shm_write16(dev, B43_SHM_SHARED, 0x0074, 0x0000);
03092

03094         b43_adjust_opmode(dev);
03095 
03096         if (dev->dev->core_rev < 3) {
03097                 b43_write16(dev, 0x060E, 0x0000);
03098                 b43_write16(dev, 0x0610, 0x8000);
03099                 b43_write16(dev, 0x0604, 0x0000);
03100                 b43_write16(dev, 0x0606, 0x0200);
03101         } else {
03102                 b43_write32(dev, 0x0188, 0x80000000);
03103                 b43_write32(dev, 0x018C, 0x02000000);
03104         }
03105         b43_write32(dev, B43_MMIO_GEN_IRQ_REASON, 0x00004000);
03106         b43_write32(dev, B43_MMIO_DMA0_IRQ_MASK, 0x0001DC00);
03107         b43_write32(dev, B43_MMIO_DMA1_IRQ_MASK, 0x0000DC00);
03108         b43_write32(dev, B43_MMIO_DMA2_IRQ_MASK, 0x0000DC00);
03109         b43_write32(dev, B43_MMIO_DMA3_IRQ_MASK, 0x0001DC00);
03110         b43_write32(dev, B43_MMIO_DMA4_IRQ_MASK, 0x0000DC00);
03111         b43_write32(dev, B43_MMIO_DMA5_IRQ_MASK, 0x0000DC00);
03112 
03113         b43_mac_phy_clock_set(dev, true);
03114 
03115         switch (dev->dev->bus_type) {
03116 #ifdef CONFIG_B43_BCMA
03117         case B43_BUS_BCMA:
03118                 /* FIXME: 0xE74 is quite common, but should be read from CC */
03119                 b43_write16(dev, B43_MMIO_POWERUP_DELAY, 0xE74);
03120                 break;
03121 #endif
03122 #ifdef CONFIG_B43_SSB
03123         case B43_BUS_SSB:
03124                 b43_write16(dev, B43_MMIO_POWERUP_DELAY,
03125                             dev->dev->sdev->bus->chipco.fast_pwrup_delay);
03126                 break;
03127 #endif
03128         }
03129 
03130         err = 0;
03131         b43dbg(dev->wl, "Chip initialized\n");
03132 out:
03133         return err;
03134 
03135 err_gpio_clean:
03136         b43_gpio_cleanup(dev);
03137         return err;
03138 }

03216 /* Periodic work locking policy:
03217  *      The whole periodic work handler is protected by
03218  *      wl->mutex. If another lock is needed somewhere in the
03219  *      pwork callchain, it's acquired in-place, where it's needed.
03220  */
03221 static void b43_periodic_work_handler(struct work_struct *work)
03222 {
03223         struct b43_wldev *dev = container_of(work, struct b43_wldev,
03224                                              periodic_work.work);
03225         struct b43_wl *wl = dev->wl;
03226         unsigned long delay;
03227 
03228         mutex_lock(&wl->mutex);
03229 
03230         if (unlikely(b43_status(dev) != B43_STAT_STARTED))
03231                 goto out;
03232         if (b43_debug(dev, B43_DBG_PWORK_STOP))
03233                 goto out_requeue;
03234 
03235         do_periodic_work(dev);
03236 
03237         dev->periodic_state++;
03238 out_requeue:
03239         if (b43_debug(dev, B43_DBG_PWORK_FAST))
03240                 delay = msecs_to_jiffies(50);
03241         else
03242                 delay = round_jiffies_relative(HZ * 15);
03243         ieee80211_queue_delayed_work(wl->hw, &dev->periodic_work, delay);
03244 out:
03245         mutex_unlock(&wl->mutex);
03246 }

03257 /* Check if communication with the device works correctly. */
03258 static int b43_validate_chipaccess(struct b43_wldev *dev)
03259 {
03260         u32 v, backup0, backup4;
03261 
03262         backup0 = b43_shm_read32(dev, B43_SHM_SHARED, 0);
03263         backup4 = b43_shm_read32(dev, B43_SHM_SHARED, 4);
03264 
03265         /* Check for read/write and endianness problems. */
03266         b43_shm_write32(dev, B43_SHM_SHARED, 0, 0x55AAAA55);
03267         if (b43_shm_read32(dev, B43_SHM_SHARED, 0) != 0x55AAAA55)
03268                 goto error;
03269         b43_shm_write32(dev, B43_SHM_SHARED, 0, 0xAA5555AA);
03270         if (b43_shm_read32(dev, B43_SHM_SHARED, 0) != 0xAA5555AA)
03271                 goto error;
03272 
03273         /* Check if unaligned 32bit SHM_SHARED access works properly.
03274          * However, don't bail out on failure, because it's noncritical. */
03275         b43_shm_write16(dev, B43_SHM_SHARED, 0, 0x1122);
03276         b43_shm_write16(dev, B43_SHM_SHARED, 2, 0x3344);
03277         b43_shm_write16(dev, B43_SHM_SHARED, 4, 0x5566);
03278         b43_shm_write16(dev, B43_SHM_SHARED, 6, 0x7788);
03279         if (b43_shm_read32(dev, B43_SHM_SHARED, 2) != 0x55663344)
03280                 b43warn(dev->wl, "Unaligned 32bit SHM read access is broken\n");
03281         b43_shm_write32(dev, B43_SHM_SHARED, 2, 0xAABBCCDD);
03282         if (b43_shm_read16(dev, B43_SHM_SHARED, 0) != 0x1122 ||
03283             b43_shm_read16(dev, B43_SHM_SHARED, 2) != 0xCCDD ||
03284             b43_shm_read16(dev, B43_SHM_SHARED, 4) != 0xAABB ||
03285             b43_shm_read16(dev, B43_SHM_SHARED, 6) != 0x7788)
03286                 b43warn(dev->wl, "Unaligned 32bit SHM write access is broken\n");
03287 
03288         b43_shm_write32(dev, B43_SHM_SHARED, 0, backup0);
03289         b43_shm_write32(dev, B43_SHM_SHARED, 4, backup4);
03290 
03291         if ((dev->dev->core_rev >= 3) && (dev->dev->core_rev <= 10)) {
03292                 /* The 32bit register shadows the two 16bit registers
03293                  * with update sideeffects. Validate this. */
03294                 b43_write16(dev, B43_MMIO_TSF_CFP_START, 0xAAAA);
03295                 b43_write32(dev, B43_MMIO_TSF_CFP_START, 0xCCCCBBBB);
03296                 if (b43_read16(dev, B43_MMIO_TSF_CFP_START_LOW) != 0xBBBB)
03297                         goto error;
03298                 if (b43_read16(dev, B43_MMIO_TSF_CFP_START_HIGH) != 0xCCCC)
03299                         goto error;
03300         }
03301         b43_write32(dev, B43_MMIO_TSF_CFP_START, 0);
03302 
03303         v = b43_read32(dev, B43_MMIO_MACCTL);
03304         v |= B43_MACCTL_GMODE;
03305         if (v != (B43_MACCTL_GMODE | B43_MACCTL_IHR_ENABLED))
03306                 goto error;
03307 
03308         return 0;
03309 error:
03310         b43err(dev->wl, "Failed to validate the chipaccess\n");
03311         return -ENODEV;
03312 }

03423 static void b43_qos_params_upload(struct b43_wldev *dev,
03424                                   const struct ieee80211_tx_queue_params *p,
03425                                   u16 shm_offset)
03426 {
03427         u16 params[B43_NR_QOSPARAMS];
03428         int bslots, tmp;
03429         unsigned int i;
03430 
03431         if (!dev->qos_enabled)
03432                 return;
03433 
03434         bslots = b43_read16(dev, B43_MMIO_RNG) & p->cw_min;
03435 
03436         memset(&params, 0, sizeof(params));
03437 
03438         params[B43_QOSPARAM_TXOP] = p->txop * 32;
03439         params[B43_QOSPARAM_CWMIN] = p->cw_min;
03440         params[B43_QOSPARAM_CWMAX] = p->cw_max;
03441         params[B43_QOSPARAM_CWCUR] = p->cw_min;
03442         params[B43_QOSPARAM_AIFS] = p->aifs;
03443         params[B43_QOSPARAM_BSLOTS] = bslots;
03444         params[B43_QOSPARAM_REGGAP] = bslots + p->aifs;
03445 
03446         for (i = 0; i < ARRAY_SIZE(params); i++) {
03447                 if (i == B43_QOSPARAM_STATUS) {
03448                         tmp = b43_shm_read16(dev, B43_SHM_SHARED,
03449                                              shm_offset + (i * 2));
03450                         /* Mark the parameters as updated. */
03451                         tmp |= 0x100;
03452                         b43_shm_write16(dev, B43_SHM_SHARED,
03453                                         shm_offset + (i * 2),
03454                                         tmp);
03455                 } else {
03456                         b43_shm_write16(dev, B43_SHM_SHARED,
03457                                         shm_offset + (i * 2),
03458                                         params[i]);
03459                 }
03460         }
03461 }

03538 /* Initialize the core's QOS capabilities */
03539 static void b43_qos_init(struct b43_wldev *dev)
03540 {
03541         if (!dev->qos_enabled) {
03542                 /* Disable QOS support. */
03543                 b43_hf_write(dev, b43_hf_read(dev) & ~B43_HF_EDCF);
03544                 b43_write16(dev, B43_MMIO_IFSCTL,
03545                             b43_read16(dev, B43_MMIO_IFSCTL)
03546                             & ~B43_MMIO_IFSCTL_USE_EDCF);
03547                 b43dbg(dev->wl, "QoS disabled\n");

03552         b43_qos_upload_all(dev);
03553 
03554         /* Enable QOS support. */
03555         b43_hf_write(dev, b43_hf_read(dev) | B43_HF_EDCF);
03556         b43_write16(dev, B43_MMIO_IFSCTL,
03557                     b43_read16(dev, B43_MMIO_IFSCTL)
03558                     | B43_MMIO_IFSCTL_USE_EDCF);
03559         b43dbg(dev->wl, "QoS enabled\n");
03560 }

03689 /* Expects wl->mutex locked */
03690 static int b43_switch_band(struct b43_wl *wl, struct ieee80211_channel *chan)
03691 {
03692         struct b43_wldev *up_dev = NULL;
03693         struct b43_wldev *down_dev;
03694         struct b43_wldev *d;
03695         int err;
03696         bool uninitialized_var(gmode);
03697         int prev_status;
03698 
03699         /* Find a device and PHY which supports the band. */
03700         list_for_each_entry(d, &wl->devlist, list) {
03701                 switch (chan->band) {
03702                 case IEEE80211_BAND_5GHZ:
03703                         if (d->phy.supports_5ghz) {
03704                                 up_dev = d;
03705                                 gmode = 0;
03706                         }
03707                         break;
03708                 case IEEE80211_BAND_2GHZ:
03709                         if (d->phy.supports_2ghz) {
03710                                 up_dev = d;
03711                                 gmode = 1;
03712                         }
03713                         break;
03714                 default:
03715                         B43_WARN_ON(1);
03716                         return -EINVAL;
03717                 }
03718                 if (up_dev)
03719                         break;
03720         }
03721         if (!up_dev) {
03722                 b43err(wl, "Could not find a device for %s-GHz band operation\n",
03723                        band_to_string(chan->band));
03724                 return -ENODEV;
03725         }
03726         if ((up_dev == wl->current_dev) &&
03727             (!!wl->current_dev->phy.gmode == !!gmode)) {
03728                 /* This device is already running. */
03729                 return 0;
03730         }
03731         b43dbg(wl, "Switching to %s-GHz band\n",
03732                band_to_string(chan->band));
03733         down_dev = wl->current_dev;
03734 
03735         prev_status = b43_status(down_dev);
03736         /* Shutdown the currently running core. */
03737         if (prev_status >= B43_STAT_STARTED)
03738                 down_dev = b43_wireless_core_stop(down_dev);
03739         if (prev_status >= B43_STAT_INITIALIZED)
03740                 b43_wireless_core_exit(down_dev);
03741 
03742         if (down_dev != up_dev) {
03743                 /* We switch to a different core, so we put PHY into
03744                  * RESET on the old core. */
03745                 b43_put_phy_into_reset(down_dev);
03746         }
03747 
03748         /* Now start the new core. */
03749         up_dev->phy.gmode = gmode;
03750         if (prev_status >= B43_STAT_INITIALIZED) {
03751                 err = b43_wireless_core_init(up_dev);
03752                 if (err) {
03753                         b43err(wl, "Fatal: Could not initialize device for "
03754                                "selected %s-GHz band\n",
03755                                band_to_string(chan->band));
03756                         goto init_failure;
03757                 }
03758         }
03759         if (prev_status >= B43_STAT_STARTED) {
03760                 err = b43_wireless_core_start(up_dev);
03761                 if (err) {
03762                         b43err(wl, "Fatal: Coult not start device for "
03763                                "selected %s-GHz band\n",
03764                                band_to_string(chan->band));
03765                         b43_wireless_core_exit(up_dev);
03766                         goto init_failure;
03767                 }
03768         }
03769         B43_WARN_ON(b43_status(up_dev) != prev_status);
03770 
03771         wl->current_dev = up_dev;
03772 
03773         return 0;
03774 init_failure:
03775         /* Whoops, failed to init the new core. No core is operating now. */
03776         wl->current_dev = NULL;
03777         return err;
03778 }

03780 /* Write the short and long frame retry limit values. */
03781 static void b43_set_retry_limits(struct b43_wldev *dev,
03782                                  unsigned int short_retry,
03783                                  unsigned int long_retry)
03784 {
03785         /* The retry limit is a 4-bit counter. Enforce this to avoid overflowing
03786          * the chip-internal counter. */
03787         short_retry = min(short_retry, (unsigned int)0xF);
03788         long_retry = min(long_retry, (unsigned int)0xF);
03789 
03790         b43_shm_write16(dev, B43_SHM_SCRATCH, B43_SHM_SC_SRLIMIT,
03791                         short_retry);
03792         b43_shm_write16(dev, B43_SHM_SCRATCH, B43_SHM_SC_LRLIMIT,
03793                         long_retry);
03794 }

03796 static int b43_op_config(struct ieee80211_hw *hw, u32 changed)
03797 {
03798         struct b43_wl *wl = hw_to_b43_wl(hw);
03799         struct b43_wldev *dev;
03800         struct b43_phy *phy;
03801         struct ieee80211_conf *conf = &hw->conf;
03802         int antenna;
03803         int err = 0;
03804         bool reload_bss = false;
03805 
03806         mutex_lock(&wl->mutex);
03807 
03808         dev = wl->current_dev;
03809 
03810         /* Switch the band (if necessary). This might change the active core. */
03811         err = b43_switch_band(wl, conf->channel);
03812         if (err)
03813                 goto out_unlock_mutex;
03814 
03815         /* Need to reload all settings if the core changed */
03816         if (dev != wl->current_dev) {
03817                 dev = wl->current_dev;
03818                 changed = ~0;
03819                 reload_bss = true;
03820         }
03821 
03822         phy = &dev->phy;
03823 
03824         if (conf_is_ht(conf))
03825                 phy->is_40mhz =
03826                         (conf_is_ht40_minus(conf) || conf_is_ht40_plus(conf));
03827         else
03828                 phy->is_40mhz = false;
03829 
03830         b43_mac_suspend(dev);
03831 
03832         if (changed & IEEE80211_CONF_CHANGE_RETRY_LIMITS)
03833                 b43_set_retry_limits(dev, conf->short_frame_max_tx_count,
03834                                           conf->long_frame_max_tx_count);
03835         changed &= ~IEEE80211_CONF_CHANGE_RETRY_LIMITS;
03836         if (!changed)
03837                 goto out_mac_enable;
03838 
03839         /* Switch to the requested channel.
03840          * The firmware takes care of races with the TX handler. */
03841         if (conf->channel->hw_value != phy->channel)
03842                 b43_switch_channel(dev, conf->channel->hw_value);
03843 
03844         dev->wl->radiotap_enabled = !!(conf->flags & IEEE80211_CONF_MONITOR);
03845 
03846         /* Adjust the desired TX power level. */
03847         if (conf->power_level != 0) {
03848                 if (conf->power_level != phy->desired_txpower) {
03849                         phy->desired_txpower = conf->power_level;
03850                         b43_phy_txpower_check(dev, B43_TXPWR_IGNORE_TIME |
03851                                                    B43_TXPWR_IGNORE_TSSI);
03852                 }
03853         }
03854 
03855         /* Antennas for RX and management frame TX. */
03856         antenna = B43_ANTENNA_DEFAULT;
03857         b43_mgmtframe_txantenna(dev, antenna);
03858         antenna = B43_ANTENNA_DEFAULT;
03859         if (phy->ops->set_rx_antenna)
03860                 phy->ops->set_rx_antenna(dev, antenna);
03861 
03862         if (wl->radio_enabled != phy->radio_on) {
03863                 if (wl->radio_enabled) {
03864                         b43_software_rfkill(dev, false);
03865                         b43info(dev->wl, "Radio turned on by software\n");
03866                         if (!dev->radio_hw_enable) {
03867                                 b43info(dev->wl, "The hardware RF-kill button "
03868                                         "still turns the radio physically off. "
03869                                         "Press the button to turn it on.\n");
03870                         }
03871                 } else {
03872                         b43_software_rfkill(dev, true);
03873                         b43info(dev->wl, "Radio turned off by software\n");
03874                 }
03875         }
03876 
03877 out_mac_enable:
03878         b43_mac_enable(dev);
03879 out_unlock_mutex:
03880         mutex_unlock(&wl->mutex);
03881 
03882         if (wl->vif && reload_bss)
03883                 b43_op_bss_info_changed(hw, wl->vif, &wl->vif->bss_conf, ~0);
03884 
03885         return err;
03886 }

03992 static int b43_op_set_key(struct ieee80211_hw *hw, enum set_key_cmd cmd,
03993                           struct ieee80211_vif *vif, struct ieee80211_sta *sta,
03994                           struct ieee80211_key_conf *key)
03995 {
03996         struct b43_wl *wl = hw_to_b43_wl(hw);
03997         struct b43_wldev *dev;
03998         u8 algorithm;
03999         u8 index;
04000         int err;
04001         static const u8 bcast_addr[ETH_ALEN] = { 0xff, 0xff, 0xff, 0xff, 0xff, 0xff };
04002 
04003         if (modparam_nohwcrypt)
04004                 return -ENOSPC; /* User disabled HW-crypto */
04005 
04006         mutex_lock(&wl->mutex);
04007 
04008         dev = wl->current_dev;
04009         err = -ENODEV;
04010         if (!dev || b43_status(dev) < B43_STAT_INITIALIZED)
04011                 goto out_unlock;
04012 
04013         if (dev->fw.pcm_request_failed || !dev->hwcrypto_enabled) {
04014                 /* We don't have firmware for the crypto engine.
04015                  * Must use software-crypto. */
04016                 err = -EOPNOTSUPP;
04017                 goto out_unlock;
04018         }
04019 
04020         err = -EINVAL;
04021         switch (key->cipher) {
04022         case WLAN_CIPHER_SUITE_WEP40:
04023                 algorithm = B43_SEC_ALGO_WEP40;
04024                 break;
04025         case WLAN_CIPHER_SUITE_WEP104:
04026                 algorithm = B43_SEC_ALGO_WEP104;
04027                 break;
04028         case WLAN_CIPHER_SUITE_TKIP:
04029                 algorithm = B43_SEC_ALGO_TKIP;
04030                 break;
04031         case WLAN_CIPHER_SUITE_CCMP:
04032                 algorithm = B43_SEC_ALGO_AES;
04033                 break;
04034         default:
04035                 B43_WARN_ON(1);
04036                 goto out_unlock;
04037         }
04038         index = (u8) (key->keyidx);
04039         if (index > 3)
04040                 goto out_unlock;
04041 
04042         switch (cmd) {
04043         case SET_KEY:
04044                 if (algorithm == B43_SEC_ALGO_TKIP &&
04045                     (!(key->flags & IEEE80211_KEY_FLAG_PAIRWISE) ||
04046                     !modparam_hwtkip)) {
04047                         /* We support only pairwise key */
04048                         err = -EOPNOTSUPP;
04049                         goto out_unlock;
04050                 }
04051 
04052                 if (key->flags & IEEE80211_KEY_FLAG_PAIRWISE) {
04053                         if (WARN_ON(!sta)) {
04054                                 err = -EOPNOTSUPP;
04055                                 goto out_unlock;
04056                         }
04057                         /* Pairwise key with an assigned MAC address. */
04058                         err = b43_key_write(dev, -1, algorithm,
04059                                             key->key, key->keylen,
04060                                             sta->addr, key);
04061                 } else {
04062                         /* Group key */
04063                         err = b43_key_write(dev, index, algorithm,
04064                                             key->key, key->keylen, NULL, key);
04065                 }
04066                 if (err)
04067                         goto out_unlock;
04068 
04069                 if (algorithm == B43_SEC_ALGO_WEP40 ||
04070                     algorithm == B43_SEC_ALGO_WEP104) {
04071                         b43_hf_write(dev, b43_hf_read(dev) | B43_HF_USEDEFKEYS);
04072                 } else {
04073                         b43_hf_write(dev,
04074                                      b43_hf_read(dev) & ~B43_HF_USEDEFKEYS);
04075                 }
04076                 key->flags |= IEEE80211_KEY_FLAG_GENERATE_IV;
04077                 if (algorithm == B43_SEC_ALGO_TKIP)
04078                         key->flags |= IEEE80211_KEY_FLAG_GENERATE_MMIC;
04079                 break;
04080         case DISABLE_KEY: {
04081                 err = b43_key_clear(dev, key->hw_key_idx);
04082                 if (err)
04083                         goto out_unlock;
04084                 break;
04085         }
04086         default:
04087                 B43_WARN_ON(1);
04088         }
04089 
04090 out_unlock:
04091         if (!err) {
04092                 b43dbg(wl, "%s hardware based encryption for keyidx: %d, "
04093                        "mac: %pM\n",
04094                        cmd == SET_KEY ? "Using" : "Disabling", key->keyidx,
04095                        sta ? sta->addr : bcast_addr);
04096                 b43_dump_keymemory(dev);
04097         }
04098         mutex_unlock(&wl->mutex);
04099 
04100         return err;
04101 }

04142 /* Locking: wl->mutex
04143  * Returns the current dev. This might be different from the passed in dev,
04144  * because the core might be gone away while we unlocked the mutex. */
04145 static struct b43_wldev * b43_wireless_core_stop(struct b43_wldev *dev)
04146 {
04147         struct b43_wl *wl;
04148         struct b43_wldev *orig_dev;
04149         u32 mask;
04150 
04151         if (!dev)
04152                 return NULL;
04153         wl = dev->wl;
04154 redo:
04155         if (!dev || b43_status(dev) < B43_STAT_STARTED)
04156                 return dev;
04157 
04158         /* Cancel work. Unlock to avoid deadlocks. */
04159         mutex_unlock(&wl->mutex);
04160         cancel_delayed_work_sync(&dev->periodic_work);
04161         cancel_work_sync(&wl->tx_work);
04162         mutex_lock(&wl->mutex);
04163         dev = wl->current_dev;
04164         if (!dev || b43_status(dev) < B43_STAT_STARTED) {
04165                 /* Whoops, aliens ate up the device while we were unlocked. */
04166                 return dev;
04167         }
04168 
04169         /* Disable interrupts on the device. */
04170         b43_set_status(dev, B43_STAT_INITIALIZED);
04171         if (b43_bus_host_is_sdio(dev->dev)) {
04172                 /* wl->mutex is locked. That is enough. */
04173                 b43_write32(dev, B43_MMIO_GEN_IRQ_MASK, 0);
04174                 b43_read32(dev, B43_MMIO_GEN_IRQ_MASK); /* Flush */
04175         } else {
04176                 spin_lock_irq(&wl->hardirq_lock);
04177                 b43_write32(dev, B43_MMIO_GEN_IRQ_MASK, 0);
04178                 b43_read32(dev, B43_MMIO_GEN_IRQ_MASK); /* Flush */
04179                 spin_unlock_irq(&wl->hardirq_lock);
04180         }
04181         /* Synchronize and free the interrupt handlers. Unlock to avoid deadlocks. */
04182         orig_dev = dev;
04183         mutex_unlock(&wl->mutex);
04184         if (b43_bus_host_is_sdio(dev->dev)) {
04185                 b43_sdio_free_irq(dev);
04186         } else {
04187 #if LINUX_VERSION_CODE < KERNEL_VERSION(2,6,31)
04188                 compat_synchronize_threaded_irq(&dev->irq_compat);
04189                 compat_free_threaded_irq(&dev->irq_compat);
04190 #else
04191                 synchronize_irq(dev->dev->irq);
04192                 free_irq(dev->dev->irq, dev);
04193 #endif
04194         }
04195         mutex_lock(&wl->mutex);
04196         dev = wl->current_dev;
04197         if (!dev)
04198                 return dev;
04199         if (dev != orig_dev) {
04200                 if (b43_status(dev) >= B43_STAT_STARTED)
04201                         goto redo;
04202                 return dev;
04203         }
04204         mask = b43_read32(dev, B43_MMIO_GEN_IRQ_MASK);
04205         B43_WARN_ON(mask != 0xFFFFFFFF && mask);
04206 
04207         /* Drain the TX queue */
04208         while (skb_queue_len(&wl->tx_queue))
04209                 dev_kfree_skb(skb_dequeue(&wl->tx_queue));
04210 
04211         b43_mac_suspend(dev);
04212         b43_leds_exit(dev);
04213         b43dbg(wl, "Wireless interface stopped\n");
04214 
04215         return dev;
04216 }

04269 /* Get PHY and RADIO versioning numbers */
04270 static int b43_phy_versioning(struct b43_wldev *dev)
04271 {
04272         struct b43_phy *phy = &dev->phy;
04273         u32 tmp;
04274         u8 analog_type;
04275         u8 phy_type;
04276         u8 phy_rev;
04277         u16 radio_manuf;
04278         u16 radio_ver;
04279         u16 radio_rev;
04280         int unsupported = 0;
04281 
04282         /* Get PHY versioning */
04283         tmp = b43_read16(dev, B43_MMIO_PHY_VER);
04284         analog_type = (tmp & B43_PHYVER_ANALOG) >> B43_PHYVER_ANALOG_SHIFT;
04285         phy_type = (tmp & B43_PHYVER_TYPE) >> B43_PHYVER_TYPE_SHIFT;
04286         phy_rev = (tmp & B43_PHYVER_VERSION);
04287         switch (phy_type) {
04288         case B43_PHYTYPE_A:
04289                 if (phy_rev >= 4)
04290                         unsupported = 1;
04291                 break;
04292         case B43_PHYTYPE_B:
04293                 if (phy_rev != 2 && phy_rev != 4 && phy_rev != 6
04294                     && phy_rev != 7)
04295                         unsupported = 1;
04296                 break;
04297         case B43_PHYTYPE_G:
04298                 if (phy_rev > 9)
04299                         unsupported = 1;
04300                 break;
04301 #ifdef CONFIG_B43_PHY_N
04302         case B43_PHYTYPE_N:
04303                 if (phy_rev > 9)
04304                         unsupported = 1;
04305                 break;
04306 #endif
04307 #ifdef CONFIG_B43_PHY_LP
04308         case B43_PHYTYPE_LP:
04309                 if (phy_rev > 2)
04310                         unsupported = 1;
04311                 break;
04312 #endif
04313 #ifdef CONFIG_B43_PHY_HT
04314         case B43_PHYTYPE_HT:
04315                 if (phy_rev > 1)
04316                         unsupported = 1;
04317                 break;
04318 #endif
04319 #ifdef CONFIG_B43_PHY_LCN
04320         case B43_PHYTYPE_LCN:
04321                 if (phy_rev > 1)
04322                         unsupported = 1;
04323                 break;
04324 #endif
04325         default:
04326                 unsupported = 1;
04327         }
04328         if (unsupported) {
04329                 b43err(dev->wl, "FOUND UNSUPPORTED PHY "
04330                        "(Analog %u, Type %u, Revision %u)\n",
04331                        analog_type, phy_type, phy_rev);
04332                 return -EOPNOTSUPP;
04333         }
04334         b43dbg(dev->wl, "Found PHY: Analog %u, Type %u, Revision %u\n",
04335                analog_type, phy_type, phy_rev);
04336 
04337         /* Get RADIO versioning */
04338         if (dev->dev->core_rev >= 24) {
04339                 u16 radio24[3];
04340 
04341                 for (tmp = 0; tmp < 3; tmp++) {
04342                         b43_write16(dev, B43_MMIO_RADIO24_CONTROL, tmp);
04343                         radio24[tmp] = b43_read16(dev, B43_MMIO_RADIO24_DATA);
04344                 }
04345 
04346                 /* Broadcom uses "id" for our "ver" and has separated "ver" */
04347                 /* radio_ver = (radio24[0] & 0xF0) >> 4; */
04348 
04349                 radio_manuf = 0x17F;
04350                 radio_ver = (radio24[2] << 8) | radio24[1];
04351                 radio_rev = (radio24[0] & 0xF);
04352         } else {
04353                 if (dev->dev->chip_id == 0x4317) {
04354                         if (dev->dev->chip_rev == 0)
04355                                 tmp = 0x3205017F;
04356                         else if (dev->dev->chip_rev == 1)
04357                                 tmp = 0x4205017F;
04358                         else
04359                                 tmp = 0x5205017F;
04360                 } else {
04361                         b43_write16(dev, B43_MMIO_RADIO_CONTROL,
04362                                     B43_RADIOCTL_ID);
04363                         tmp = b43_read16(dev, B43_MMIO_RADIO_DATA_LOW);
04364                         b43_write16(dev, B43_MMIO_RADIO_CONTROL,
04365                                     B43_RADIOCTL_ID);
04366                         tmp |= (u32)b43_read16(dev, B43_MMIO_RADIO_DATA_HIGH)
04367                                 << 16;
04368                 }
04369                 radio_manuf = (tmp & 0x00000FFF);
04370                 radio_ver = (tmp & 0x0FFFF000) >> 12;
04371                 radio_rev = (tmp & 0xF0000000) >> 28;
04372         }
04373 
04374         if (radio_manuf != 0x17F /* Broadcom */)
04375                 unsupported = 1;
04376         switch (phy_type) {
04377         case B43_PHYTYPE_A:
04378                 if (radio_ver != 0x2060)
04379                         unsupported = 1;
04380                 if (radio_rev != 1)
04381                         unsupported = 1;
04382                 if (radio_manuf != 0x17F)
04383                         unsupported = 1;
04384                 break;
04385         case B43_PHYTYPE_B:
04386                 if ((radio_ver & 0xFFF0) != 0x2050)
04387                         unsupported = 1;
04388                 break;
04389         case B43_PHYTYPE_G:
04390                 if (radio_ver != 0x2050)
04391                         unsupported = 1;
04392                 break;
04393         case B43_PHYTYPE_N:
04394                 if (radio_ver != 0x2055 && radio_ver != 0x2056)
04395                         unsupported = 1;
04396                 break;
04397         case B43_PHYTYPE_LP:
04398                 if (radio_ver != 0x2062 && radio_ver != 0x2063)
04399                         unsupported = 1;
04400                 break;
04401         case B43_PHYTYPE_HT:
04402                 if (radio_ver != 0x2059)
04403                         unsupported = 1;
04404                 break;
04405         case B43_PHYTYPE_LCN:
04406                 if (radio_ver != 0x2064)
04407                         unsupported = 1;
04408                 break;
04409         default:
04410                 B43_WARN_ON(1);
04411         }
04412         if (unsupported) {
04413                 b43err(dev->wl, "FOUND UNSUPPORTED RADIO "
04414                        "(Manuf 0x%X, Version 0x%X, Revision %u)\n",
04415                        radio_manuf, radio_ver, radio_rev);
04416                 return -EOPNOTSUPP;
04417         }
04418         b43dbg(dev->wl, "Found Radio: Manuf 0x%X, Version 0x%X, Revision %u\n",
04419                radio_manuf, radio_ver, radio_rev);
04420 
04421         phy->radio_manuf = radio_manuf;
04422         phy->radio_ver = radio_ver;
04423         phy->radio_rev = radio_rev;
04424 
04425         phy->analog = analog_type;
04426         phy->type = phy_type;
04427         phy->rev = phy_rev;
04428 
04429         return 0;
04430 }

04555 /* Shutdown a wireless core */
04556 /* Locking: wl->mutex */
04557 static void b43_wireless_core_exit(struct b43_wldev *dev)
04558 {
04559         u32 macctl;
04560 
04561         B43_WARN_ON(dev && b43_status(dev) > B43_STAT_INITIALIZED);
04562         if (!dev || b43_status(dev) != B43_STAT_INITIALIZED)
04563                 return;
04564 
04565         /* Unregister HW RNG driver */
04566         b43_rng_exit(dev->wl);
04567 
04568         b43_set_status(dev, B43_STAT_UNINIT);
04569 
04570         /* Stop the microcode PSM. */
04571         macctl = b43_read32(dev, B43_MMIO_MACCTL);
04572         macctl &= ~B43_MACCTL_PSM_RUN;
04573         macctl |= B43_MACCTL_PSM_JMP0;
04574         b43_write32(dev, B43_MMIO_MACCTL, macctl);
04575 
04576         b43_dma_free(dev);
04577         b43_pio_free(dev);
04578         b43_chip_exit(dev);
04579         dev->phy.ops->switch_analog(dev, 0);
04580         if (dev->wl->current_beacon) {
04581                 dev_kfree_skb_any(dev->wl->current_beacon);
04582                 dev->wl->current_beacon = NULL;
04583         }
04584 
04585         b43_device_disable(dev, 0);
04586         b43_bus_may_powerdown(dev);

04589 /* Initialize a wireless core */
04590 static int b43_wireless_core_init(struct b43_wldev *dev)
04591 {
04592         struct ssb_sprom *sprom = dev->dev->bus_sprom;
04593         struct b43_phy *phy = &dev->phy;
04594         int err;
04595         u64 hf;
04596 
04597         B43_WARN_ON(b43_status(dev) != B43_STAT_UNINIT);
04598 
04599         err = b43_bus_powerup(dev, 0);
04600         if (err)
04601                 goto out;
04602         if (!b43_device_is_enabled(dev))
04603                 b43_wireless_core_reset(dev, phy->gmode);
04604 
04605         /* Reset all data structures. */
04606         setup_struct_wldev_for_init(dev);
04607         phy->ops->prepare_structs(dev);
04608 
04609         /* Enable IRQ routing to this device. */
04610         switch (dev->dev->bus_type) {
04611 #ifdef CONFIG_B43_BCMA
04612         case B43_BUS_BCMA:
04613                 bcma_core_pci_irq_ctl(&dev->dev->bdev->bus->drv_pci,
04614                                       dev->dev->bdev, true);
04615                 break;
04616 #endif
04617 #ifdef CONFIG_B43_SSB
04618         case B43_BUS_SSB:
04619                 ssb_pcicore_dev_irqvecs_enable(&dev->dev->sdev->bus->pcicore,
04620                                                dev->dev->sdev);
04621                 break;
04622 #endif
04623         }
04624 
04625         b43_imcfglo_timeouts_workaround(dev);
04626         b43_bluetooth_coext_disable(dev);
04627         if (phy->ops->prepare_hardware) {
04628                 err = phy->ops->prepare_hardware(dev);
04629                 if (err)
04630                         goto err_busdown;
04631         }
04632         err = b43_chip_init(dev);
04633         if (err)
04634                 goto err_busdown;
04635         b43_shm_write16(dev, B43_SHM_SHARED,
04636                         B43_SHM_SH_WLCOREREV, dev->dev->core_rev);
04637         hf = b43_hf_read(dev);
04638         if (phy->type == B43_PHYTYPE_G) {
04639                 hf |= B43_HF_SYMW;
04640                 if (phy->rev == 1)
04641                         hf |= B43_HF_GDCW;
04642                 if (sprom->boardflags_lo & B43_BFL_PACTRL)
04643                         hf |= B43_HF_OFDMPABOOST;
04644         }
04645         if (phy->radio_ver == 0x2050) {
04646                 if (phy->radio_rev == 6)
04647                         hf |= B43_HF_4318TSSI;
04648                 if (phy->radio_rev < 6)
04649                         hf |= B43_HF_VCORECALC;
04650         }
04651         if (sprom->boardflags_lo & B43_BFL_XTAL_NOSLOW)
04652                 hf |= B43_HF_DSCRQ; /* Disable slowclock requests from ucode. */
04653 #ifdef CONFIG_SSB_DRIVER_PCICORE
04654         if (dev->dev->bus_type == B43_BUS_SSB &&
04655             dev->dev->sdev->bus->bustype == SSB_BUSTYPE_PCI &&
04656             dev->dev->sdev->bus->pcicore.dev->id.revision <= 10)
04657                 hf |= B43_HF_PCISCW; /* PCI slow clock workaround. */
04658 #endif
04659         hf &= ~B43_HF_SKCFPUP;
04660         b43_hf_write(dev, hf);
04661 
04662         b43_set_retry_limits(dev, B43_DEFAULT_SHORT_RETRY_LIMIT,
04663                              B43_DEFAULT_LONG_RETRY_LIMIT);
04664         b43_shm_write16(dev, B43_SHM_SHARED, B43_SHM_SH_SFFBLIM, 3);
04665         b43_shm_write16(dev, B43_SHM_SHARED, B43_SHM_SH_LFFBLIM, 2);
04666 
04667         /* Disable sending probe responses from firmware.
04668          * Setting the MaxTime to one usec will always trigger
04669          * a timeout, so we never send any probe resp.
04670          * A timeout of zero is infinite. */
04671         b43_shm_write16(dev, B43_SHM_SHARED, B43_SHM_SH_PRMAXTIME, 1);
04672 
04673         b43_rate_memory_init(dev);
04674         b43_set_phytxctl_defaults(dev);
04675 
04676         /* Minimum Contention Window */
04677         if (phy->type == B43_PHYTYPE_B)
04678                 b43_shm_write16(dev, B43_SHM_SCRATCH, B43_SHM_SC_MINCONT, 0x1F);
04679         else
04680                 b43_shm_write16(dev, B43_SHM_SCRATCH, B43_SHM_SC_MINCONT, 0xF);
04681         /* Maximum Contention Window */
04682         b43_shm_write16(dev, B43_SHM_SCRATCH, B43_SHM_SC_MAXCONT, 0x3FF);
04683 
04684         if (b43_bus_host_is_pcmcia(dev->dev) ||
04685             b43_bus_host_is_sdio(dev->dev)) {
04686                 dev->__using_pio_transfers = 1;
04687                 err = b43_pio_init(dev);
04688         } else if (dev->use_pio) {
04689                 b43warn(dev->wl, "Forced PIO by use_pio module parameter. "
04690                         "This should not be needed and will result in lower "
04691                         "performance.\n");
04692                 dev->__using_pio_transfers = 1;
04693                 err = b43_pio_init(dev);
04694         } else {
04695                 dev->__using_pio_transfers = 0;
04696                 err = b43_dma_init(dev);
04697         }
04698         if (err)
04699                 goto err_chip_exit;
04700         b43_qos_init(dev);
04701         b43_set_synth_pu_delay(dev, 1);
04702         b43_bluetooth_coext_enable(dev);
04703 
04704         b43_bus_powerup(dev, !(sprom->boardflags_lo & B43_BFL_XTAL_NOSLOW));
04705         b43_upload_card_macaddress(dev);
04706         b43_security_init(dev);
04707 
04708         ieee80211_wake_queues(dev->wl->hw);
04709 
04710         b43_set_status(dev, B43_STAT_INITIALIZED);
04711 
04712         /* Register HW RNG driver */
04713         b43_rng_init(dev->wl);
04714 
04715 out:
04716         return err;
04717 
04718 err_chip_exit:
04719         b43_chip_exit(dev);
04720 err_busdown:
04721         b43_bus_may_powerdown(dev);
04722         B43_WARN_ON(b43_status(dev) != B43_STAT_UNINIT);
04723         return err;
04724 }

04953 /* Hard-reset the chip. Do not call this directly.
04954  * Use b43_controller_restart()
04955  */
04956 static void b43_chip_reset(struct work_struct *work)
04957 {
04958         struct b43_wldev *dev =
04959             container_of(work, struct b43_wldev, restart_work);
04960         struct b43_wl *wl = dev->wl;
04961         int err = 0;
04962         int prev_status;
04963 
04964         mutex_lock(&wl->mutex);
04965 
04966         prev_status = b43_status(dev);
04967         /* Bring the device down... */
04968         if (prev_status >= B43_STAT_STARTED) {
04969                 dev = b43_wireless_core_stop(dev);
04970                 if (!dev) {
04971                         err = -ENODEV;
04972                         goto out;
04973                 }
04974         }
04975         if (prev_status >= B43_STAT_INITIALIZED)
04976                 b43_wireless_core_exit(dev);
04977 
04978         /* ...and up again. */
04979         if (prev_status >= B43_STAT_INITIALIZED) {
04980                 err = b43_wireless_core_init(dev);
04981                 if (err)
04982                         goto out;
04983         }
04984         if (prev_status >= B43_STAT_STARTED) {
04985                 err = b43_wireless_core_start(dev);
04986                 if (err) {
04987                         b43_wireless_core_exit(dev);
04988                         goto out;
04989                 }
04990         }
04991 out:
04992         if (err)
04993                 wl->current_dev = NULL; /* Failed to init the dev. */
04994         mutex_unlock(&wl->mutex);
04995 
04996         if (err) {
04997                 b43err(wl, "Controller restart FAILED\n");
04998                 return;
04999         }
05000 
05001         /* reload configuration */
05002         b43_op_config(wl->hw, ~0);
05003         if (wl->vif)
05004                 b43_op_bss_info_changed(wl->hw, wl->vif, &wl->vif->bss_conf, ~0);
05005 
05006         b43info(wl, "Controller restarted\n");
05007 }

05042 static int b43_wireless_core_attach(struct b43_wldev *dev)
05043 {
05044         struct b43_wl *wl = dev->wl;
05045         struct pci_dev *pdev = NULL;
05046         int err;
05047         u32 tmp;
05048         bool have_2ghz_phy = 0, have_5ghz_phy = 0;
05049 
05050         /* Do NOT do any device initialization here.
05051          * Do it in wireless_core_init() instead.
05052          * This function is for gathering basic information about the HW, only.
05053          * Also some structs may be set up here. But most likely you want to have
05054          * that in core_init(), too.
05055          */
05056 
05057 #ifdef CONFIG_B43_SSB
05058         if (dev->dev->bus_type == B43_BUS_SSB &&
05059             dev->dev->sdev->bus->bustype == SSB_BUSTYPE_PCI)
05060                 pdev = dev->dev->sdev->bus->host_pci;
05061 #endif
05062 
05063         err = b43_bus_powerup(dev, 0);
05064         if (err) {
05065                 b43err(wl, "Bus powerup failed\n");
05066                 goto out;
05067         }
05068 
05069         /* Get the PHY type. */
05070         switch (dev->dev->bus_type) {
05071 #ifdef CONFIG_B43_BCMA
05072         case B43_BUS_BCMA:
05073                 tmp = bcma_aread32(dev->dev->bdev, BCMA_IOST);
05074                 have_2ghz_phy = !!(tmp & B43_BCMA_IOST_2G_PHY);
05075                 have_5ghz_phy = !!(tmp & B43_BCMA_IOST_5G_PHY);
05076                 break;
05077 #endif
05078 #ifdef CONFIG_B43_SSB
05079         case B43_BUS_SSB:
05080                 if (dev->dev->core_rev >= 5) {
05081                         tmp = ssb_read32(dev->dev->sdev, SSB_TMSHIGH);
05082                         have_2ghz_phy = !!(tmp & B43_TMSHIGH_HAVE_2GHZ_PHY);
05083                         have_5ghz_phy = !!(tmp & B43_TMSHIGH_HAVE_5GHZ_PHY);
05084                 } else
05085                         B43_WARN_ON(1);
05086                 break;
05087 #endif
05088         }
05089 
05090         dev->phy.gmode = have_2ghz_phy;
05091         dev->phy.radio_on = 1;
05092         b43_wireless_core_reset(dev, dev->phy.gmode);
05093 
05094         err = b43_phy_versioning(dev);
05095         if (err)
05096                 goto err_powerdown;
05097         /* Check if this device supports multiband. */
05098         if (!pdev ||
05099             (pdev->device != 0x4312 &&
05100              pdev->device != 0x4319 && pdev->device != 0x4324)) {
05101                 /* No multiband support. */
05102                 have_2ghz_phy = 0;
05103                 have_5ghz_phy = 0;
05104                 switch (dev->phy.type) {
05105                 case B43_PHYTYPE_A:
05106                         have_5ghz_phy = 1;
05107                         break;
05108                 case B43_PHYTYPE_LP: //FIXME not always!
05109 #if 0 //FIXME enabling 5GHz causes a NULL pointer dereference
05110                         have_5ghz_phy = 1;
05111 #endif
05112                 case B43_PHYTYPE_G:
05113                 case B43_PHYTYPE_N:
05114                 case B43_PHYTYPE_HT:
05115                 case B43_PHYTYPE_LCN:
05116                         have_2ghz_phy = 1;
05117                         break;
05118                 default:
05119                         B43_WARN_ON(1);
05120                 }
05121         }
05122         if (dev->phy.type == B43_PHYTYPE_A) {
05123                 /* FIXME */
05124                 b43err(wl, "IEEE 802.11a devices are unsupported\n");
05125                 err = -EOPNOTSUPP;
05126                 goto err_powerdown;
05127         }
05128         if (1 /* disable A-PHY */) {
05129                 /* FIXME: For now we disable the A-PHY on multi-PHY devices. */
05130                 if (dev->phy.type != B43_PHYTYPE_N &&
05131                     dev->phy.type != B43_PHYTYPE_LP) {
05132                         have_2ghz_phy = 1;
05133                         have_5ghz_phy = 0;
05134                 }
05135         }
05136 
05137         err = b43_phy_allocate(dev);
05138         if (err)
05139                 goto err_powerdown;
05140 
05141         dev->phy.gmode = have_2ghz_phy;
05142         b43_wireless_core_reset(dev, dev->phy.gmode);
05143 
05144         err = b43_validate_chipaccess(dev);
05145         if (err)
05146                 goto err_phy_free;
05147         err = b43_setup_bands(dev, have_2ghz_phy, have_5ghz_phy);
05148         if (err)
05149                 goto err_phy_free;
05150 
05151         /* Now set some default "current_dev" */
05152         if (!wl->current_dev)
05153                 wl->current_dev = dev;
05154         INIT_WORK(&dev->restart_work, b43_chip_reset);
05155 
05156         dev->phy.ops->switch_analog(dev, 0);
05157         b43_device_disable(dev, 0);
05158         b43_bus_may_powerdown(dev);
05159 
05160 out:
05161         return err;
05162 
05163 err_phy_free:
05164         b43_phy_free(dev);
05165 err_powerdown:
05166         b43_bus_may_powerdown(dev);
05167         return err;
05168 }
