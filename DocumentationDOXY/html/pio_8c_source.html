00037 static u16 generate_cookie(struct b43_pio_txqueue *q,
00038                            struct b43_pio_txpacket *pack)
00039 {
00040         u16 cookie;
00041 
00042         /* Use the upper 4 bits of the cookie as
00043          * PIO controller ID and store the packet index number
00044          * in the lower 12 bits.
00045          * Note that the cookie must never be 0, as this
00046          * is a special value used in RX path.
00047          * It can also not be 0xFFFF because that is special
00048          * for multicast frames.
00049          */
00050         cookie = (((u16)q->index + 1) << 12);
00051         cookie |= pack->index;
00052 
00053         return cookie;
00054 }

00056 static
00057 struct b43_pio_txqueue *parse_cookie(struct b43_wldev *dev,
00058                                      u16 cookie,
00059                                       struct b43_pio_txpacket **pack)
00060 {
00061         struct b43_pio *pio = &dev->pio;
00062         struct b43_pio_txqueue *q = NULL;
00063         unsigned int pack_index;
00064 
00065         switch (cookie & 0xF000) {
00066         case 0x1000:
00067                 q = pio->tx_queue_AC_BK;
00068                 break;
00069         case 0x2000:
00070                 q = pio->tx_queue_AC_BE;
00071                 break;
00072         case 0x3000:
00073                 q = pio->tx_queue_AC_VI;
00074                 break;
00075         case 0x4000:
00076                 q = pio->tx_queue_AC_VO;
00077                 break;
00078         case 0x5000:
00079                 q = pio->tx_queue_mcast;
00080                 break;
00081         }
00082         if (B43_WARN_ON(!q))
00083                 return NULL;
00084         pack_index = (cookie & 0x0FFF);
00085         if (B43_WARN_ON(pack_index >= ARRAY_SIZE(q->packets)))
00086                 return NULL;
00087         *pack = &q->packets[pack_index];
00088 
00089         return q;
00090 }

00334 {

00336         struct b43_wl *wl = dev->wl;

00339         ctl |= B43_PIO_TXCTL_WRITELO | B43_PIO_TXCTL_WRITEHI;
00340         b43_piotx_write16(q, B43_PIO_TXCTL, ctl);

00343                         q->mmio_base + B43_PIO_TXDATA,

00502 int b43_pio_tx(struct b43_wldev *dev, struct sk_buff *skb)
00503 {
00504         struct b43_pio_txqueue *q;
00505         struct ieee80211_hdr *hdr;
00506         unsigned int hdrlen, total_len;
00507         int err = 0;
00508         struct ieee80211_tx_info *info = IEEE80211_SKB_CB(skb);
00509 
00510         hdr = (struct ieee80211_hdr *)skb->data;
00511 
00512         if (info->flags & IEEE80211_TX_CTL_SEND_AFTER_DTIM) {
00513                 /* The multicast queue will be sent after the DTIM. */
00514                 q = dev->pio.tx_queue_mcast;
00515                 /* Set the frame More-Data bit. Ucode will clear it
00516                  * for us on the last frame. */
00517                 hdr->frame_control |= cpu_to_le16(IEEE80211_FCTL_MOREDATA);
00518         } else {
00519                 /* Decide by priority where to put this frame. */
00520                 q = select_queue_by_priority(dev, skb_get_queue_mapping(skb));
00521         }
00522 
00523         hdrlen = b43_txhdr_size(dev);
00524         total_len = roundup(skb->len + hdrlen, 4);
00525 
00526         if (unlikely(total_len > q->buffer_size)) {
00527                 err = -ENOBUFS;
00528                 b43dbg(dev->wl, "PIO: TX packet longer than queue.\n");
00529                 goto out;
00530         }
00531         if (unlikely(q->free_packet_slots == 0)) {
00532                 err = -ENOBUFS;
00533                 b43warn(dev->wl, "PIO: TX packet overflow.\n");
00534                 goto out;
00535         }
00536         B43_WARN_ON(q->buffer_used > q->buffer_size);
00537 
00538         if (total_len > (q->buffer_size - q->buffer_used)) {
00539                 /* Not enough memory on the queue. */
00540                 err = -EBUSY;
00541                 ieee80211_stop_queue(dev->wl->hw, skb_get_queue_mapping(skb));
00542                 q->stopped = 1;
00543                 goto out;
00544         }
00545 
00546         /* Assign the queue number to the ring (if not already done before)
00547          * so TX status handling can use it. The mac80211-queue to b43-queue
00548          * mapping is static, so we don't need to store it per frame. */
00549         q->queue_prio = skb_get_queue_mapping(skb);
00550 
00551         err = pio_tx_frame(q, skb);
00552         if (unlikely(err == -ENOKEY)) {
00553                 /* Drop this packet, as we don't have the encryption key
00554                  * anymore and must not transmit it unencrypted. */
00555                 dev_kfree_skb_any(skb);
00556                 err = 0;
00557                 goto out;
00558         }
00559         if (unlikely(err)) {
00560                 b43err(dev->wl, "PIO transmission failure\n");
00561                 goto out;
00562         }
00563 
00564         B43_WARN_ON(q->buffer_used > q->buffer_size);
00565         if (((q->buffer_size - q->buffer_used) < roundup(2 + 2 + 6, 4)) ||
00566             (q->free_packet_slots == 0)) {
00567                 /* The queue is full. */
00568                 ieee80211_stop_queue(dev->wl->hw, skb_get_queue_mapping(skb));
00569                 q->stopped = 1;
00570         }
00571 
00572 out:
00573         return err;
00574 }

00621

00623         switch (dev->fw.hdr_format) {

00629         case B43_FW_HDR_598:

00656                                   B43_PIO_RXCTL_FRAMERDY);

00658                         ctl = b43_piorx_read16(q, B43_PIO_RXCTL);
00659                         if (ctl & B43_PIO_RXCTL_DATARDY)
00660                                 goto data_ready;
00661                         udelay(10);
00662                 }

00671                                q->mmio_base + B43_PIO8_RXDATA,

00692                 break;
00693         case B43_FW_HDR_410:

00703                 }

00723                         BUILD_BUG_ON(sizeof(wl->pio_tailspace) < 4);
00724 
00725                         /* Read the last few bytes. */

00728                                        sizeof(u32));

00732                                 skb->data[len + padding - 2] = tail[1];
00733                                 skb->data[len + padding - 1] = tail[2];
00734                                 break;

00736                                 skb->data[len + padding - 2] = tail[0];
00737                                 skb->data[len + padding - 1] = tail[1];
00738                                 break;
00739                         case 1:

00741                                 break;
00742                         }
00743                 }
00744         } else {

00746                                q->mmio_base + B43_PIO_RXDATA,
00747                                sizeof(u16));

00750                         BUILD_BUG_ON(sizeof(wl->pio_tailspace) < 2);
00751 
00752                         /* Read the last byte. */

00754                                        q->mmio_base + B43_PIO_RXDATA,
00755                                        sizeof(u16));

00773 }
