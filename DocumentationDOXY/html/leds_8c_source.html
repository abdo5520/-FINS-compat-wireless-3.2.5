00060 static void b43_led_update(struct b43_wldev *dev,
00061                            struct b43_led *led)
00062 {
00063         bool radio_enabled;
00064         bool turn_on;
00065 
00066         if (!led->wl)
00067                 return;
00068 
00069         radio_enabled = (dev->phy.radio_on && dev->radio_hw_enable);
00070 
00071         /* The led->state read is racy, but we don't care. In case we raced
00072          * with the brightness_set handler, we will be called again soon
00073          * to fixup our state. */
00074         if (radio_enabled)
00075                 turn_on = atomic_read(&led->state) != LED_OFF;
00076         else
00077                 turn_on = 0;
00078         if (turn_on == led->hw_state)
00079                 return;
00080         led->hw_state = turn_on;
00081 
00082         if (turn_on)
00083                 b43_led_turn_on(dev, led->index, led->activelow);
00084         else
00085                 b43_led_turn_off(dev, led->index, led->activelow);
00086 }

00213 static void b43_led_get_sprominfo(struct b43_wldev *dev,
00214                                   unsigned int led_index,
00215                                   enum b43_led_behaviour *behaviour,
00216                                   bool *activelow)
00217 {
00218         u8 sprom[4];
00219 
00220         sprom[0] = dev->dev->bus_sprom->gpio0;
00221         sprom[1] = dev->dev->bus_sprom->gpio1;
00222         sprom[2] = dev->dev->bus_sprom->gpio2;
00223         sprom[3] = dev->dev->bus_sprom->gpio3;
00224 
00225         if (sprom[led_index] == 0xFF) {
00226                 /* There is no LED information in the SPROM
00227                  * for this LED. Hardcode it here. */
00228                 *activelow = 0;
00229                 switch (led_index) {
00230                 case 0:
00231                         *behaviour = B43_LED_ACTIVITY;
00232                         *activelow = 1;
00233                         if (dev->dev->board_vendor == PCI_VENDOR_ID_COMPAQ)
00234                                 *behaviour = B43_LED_RADIO_ALL;
00235                         break;
00236                 case 1:
00237                         *behaviour = B43_LED_RADIO_B;
00238                         if (dev->dev->board_vendor == PCI_VENDOR_ID_ASUSTEK)
00239                                 *behaviour = B43_LED_ASSOC;
00240                         break;
00241                 case 2:
00242                         *behaviour = B43_LED_RADIO_A;
00243                         break;
00244                 case 3:
00245                         *behaviour = B43_LED_OFF;
00246                         break;
00247                 default:
00248                         *behaviour = B43_LED_OFF;
00249                         B43_WARN_ON(1);
00250                         return;
00251                 }
00252         } else {
00253                 *behaviour = sprom[led_index] & B43_LED_BEHAVIOUR;
00254                 *activelow = !!(sprom[led_index] & B43_LED_ACTIVELOW);
00255         }

00258 void b43_leds_init(struct b43_wldev *dev)
00259 {
00260         struct b43_led *led;
00261         unsigned int i;
00262         enum b43_led_behaviour behaviour;
00263         bool activelow;
00264 
00265         /* Sync the RF-kill LED state (if we have one) with radio and switch states. */
00266         led = &dev->wl->leds.led_radio;
00267         if (led->wl) {

00270                         led->hw_state = 1;
00271                         atomic_set(&led->state, 1);
00272                 } else {
00273                         b43_led_turn_off(dev, led->index, led->activelow);
00274                         led->hw_state = 0;
00275                         atomic_set(&led->state, 0);
00276                 }
00277         }
00278 
00279         /* Initialize TX/RX/ASSOC leds */
00280         led = &dev->wl->leds.led_tx;
00281         if (led->wl) {
00282                 b43_led_turn_off(dev, led->index, led->activelow);
00283                 led->hw_state = 0;
00284                 atomic_set(&led->state, 0);
00285         }
00286         led = &dev->wl->leds.led_rx;
00287         if (led->wl) {
00288                 b43_led_turn_off(dev, led->index, led->activelow);
00289                 led->hw_state = 0;
00290                 atomic_set(&led->state, 0);
00291         }
00292         led = &dev->wl->leds.led_assoc;
00293         if (led->wl) {
00294                 b43_led_turn_off(dev, led->index, led->activelow);
00295                 led->hw_state = 0;
00296                 atomic_set(&led->state, 0);
00297         }
00298 
00299         /* Initialize other LED states. */
00300         for (i = 0; i < B43_MAX_NR_LEDS; i++) {
00301                 b43_led_get_sprominfo(dev, i, &behaviour, &activelow);
00302                 switch (behaviour) {
00303                 case B43_LED_OFF:
00304                         b43_led_turn_off(dev, i, activelow);
00305                         break;
00306                 case B43_LED_ON:
00307                         b43_led_turn_on(dev, i, activelow);
00308                         break;
00309                 default:
00310                         /* Leave others as-is. */
00311                         break;
00312                 }
00313         }
00314 
00315         dev->wl->leds.stop = 0;
00316 }
