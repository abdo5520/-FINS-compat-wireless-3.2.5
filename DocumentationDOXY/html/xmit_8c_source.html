00148 void b43_generate_plcp_hdr(struct b43_plcp_hdr4 *plcp,
00149                            const u16 octets, const u8 bitrate)
00150 {
00151         __u8 *raw = plcp->raw;
00152 
00153         if (b43_is_ofdm_rate(bitrate)) {
00154                 u32 d;
00155 
00156                 d = b43_plcp_get_ratecode_ofdm(bitrate);
00157                 B43_WARN_ON(octets & 0xF000);
00158                 d |= (octets << 5);
00159                 plcp->data = cpu_to_le32(d);
00160         } else {
00161                 u32 plen;
00162 
00163                 plen = octets * 16 / bitrate;
00164                 if ((octets * 16 % bitrate) > 0) {
00165                         plen++;
00166                         if ((bitrate == B43_CCK_RATE_11MB)
00167                             && ((octets * 8 % 11) < 4)) {
00168                                 raw[1] = 0x84;
00169                         } else
00170                                 raw[1] = 0x04;
00171                 } else
00172                         raw[1] = 0x04;
00173                 plcp->data |= cpu_to_le32(plen << 16);
00174                 raw[0] = b43_plcp_get_ratecode_cck(bitrate);
00175         }
00176 }

00261                               phy->type == B43_PHYTYPE_N ||
00262                               phy->type == B43_PHYTYPE_HT);

00264         struct ieee80211_rate *txrate;
00265         struct ieee80211_tx_rate *rates;
00266 
00267         memset(txhdr, 0, sizeof(*txhdr));
00268 
00269         txrate = ieee80211_get_tx_rate(dev->wl->hw, info);
00270         rate = txrate ? txrate->hw_value : B43_CCK_RATE_1MB;
00271         rate_ofdm = b43_is_ofdm_rate(rate);
00272         fbrate = ieee80211_get_alt_retry_rate(dev->wl->hw, info, 0) ? : txrate;
00273         rate_fb = fbrate->hw_value;
00274         rate_fb_ofdm = b43_is_ofdm_rate(rate_fb);
00275 
00276         if (rate_ofdm)
00277                 txhdr->phy_rate = b43_plcp_get_ratecode_ofdm(rate);
00278         else
00279                 txhdr->phy_rate = b43_plcp_get_ratecode_cck(rate);
00280         txhdr->mac_frame_ctl = wlhdr->frame_control;
00281         memcpy(txhdr->tx_receiver, wlhdr->addr1, 6);
00282 
00283         /* Calculate duration for fallback rate */
00284         if ((rate_fb == rate) ||
00285             (wlhdr->duration_id & cpu_to_le16(0x8000)) ||
00286             (wlhdr->duration_id == cpu_to_le16(0))) {
00287                 /* If the fallback rate equals the normal rate or the
00288                  * dur_id field contains an AID, CFP magic or 0,
00289                  * use the original dur_id field. */
00290                 txhdr->dur_fb = wlhdr->duration_id;
00291         } else {
00292                 txhdr->dur_fb = ieee80211_generic_frame_duration(
00293                         dev->wl->hw, info->control.vif, fragment_len, fbrate);
00294         }
00295 
00296         plcp_fragment_len = fragment_len + FCS_LEN;
00297         if (use_encryption) {
00298                 u8 key_idx = info->control.hw_key->hw_key_idx;
00299                 struct b43_key *key;
00300                 int wlhdr_len;
00301                 size_t iv_len;
00302 
00303                 B43_WARN_ON(key_idx >= ARRAY_SIZE(dev->key));
00304                 key = &(dev->key[key_idx]);
00305 
00306                 if (unlikely(!key->keyconf)) {
00307                         /* This key is invalid. This might only happen
00308                          * in a short timeframe after machine resume before
00309                          * we were able to reconfigure keys.
00310                          * Drop this packet completely. Do not transmit it
00311                          * unencrypted to avoid leaking information. */
00312                         return -ENOKEY;
00313                 }
00314 
00315                 /* Hardware appends ICV. */
00316                 plcp_fragment_len += info->control.hw_key->icv_len;
00317 
00318                 key_idx = b43_kidx_to_fw(dev, key_idx);
00319                 mac_ctl |= (key_idx << B43_TXH_MAC_KEYIDX_SHIFT) &
00320                            B43_TXH_MAC_KEYIDX;

00574 static s8 b43_rssi_postprocess(struct b43_wldev *dev,
00575                                u8 in_rssi, int ofdm,
00576                                int adjust_2053, int adjust_2050)

00593                                 tmp -= 3;

00599                                 B43_WARN_ON(phy->type != B43_PHYTYPE_G);
00600                                 tmp = gphy->nrssi_lt[in_rssi];
00601                                 tmp = 31 - tmp;
00602                                 tmp *= -131;
00603                                 tmp /= 128;
00604                                 tmp -= 57;
00605                         } else {
00606                                 tmp = in_rssi;
00607                                 tmp = 31 - tmp;
00608                                 tmp *= -149;
00609                                 tmp /= 128;
00610                                 tmp -= 68;
00611                         }
00612                         if (phy->type == B43_PHYTYPE_G && adjust_2050)
00613                                 tmp += 25;
00614                 }
00615                 break;
00616         case 0x2060:

00619                 else
00620                         tmp = in_rssi;
00621                 break;

00624                 tmp -= 11;

00626                 tmp /= 64;

00681                 mactime = le16_to_cpu(rxhdr->format_351.mac_time);
00682                 chanstat = le16_to_cpu(rxhdr->format_351.channel);
00683                 break;
00684         }
00685         phytype = chanstat & B43_RX_CHAN_PHYTYPE;
00686 
00687         if (unlikely(macstat & B43_RX_MAC_FCSERR)) {
00688                 dev->wl->ieee_stats.dot11FCSErrorCount++;
00689                 status.flag |= RX_FLAG_FAILED_FCS_CRC;
00690         }
00691         if (unlikely(phystat0 & (B43_RX_PHYST0_PLCPHCF | B43_RX_PHYST0_PLCPFV)))
00692                 status.flag |= RX_FLAG_FAILED_PLCP_CRC;
00693         if (phystat0 & B43_RX_PHYST0_SHORTPRMBL)
00694                 status.flag |= RX_FLAG_SHORTPRE;
00695         if (macstat & B43_RX_MAC_DECERR) {
00696                 /* Decryption with the given key failed.
00697                  * Drop the packet. We also won't be able to decrypt it with
00698                  * the key in software. */
00699                 goto drop;
00700         }
00701 
00702         /* Skip PLCP and padding */
00703         padding = (macstat & B43_RX_MAC_PADDING) ? 2 : 0;
00704         if (unlikely(skb->len < (sizeof(struct b43_plcp_hdr6) + padding))) {
00705                 b43dbg(dev->wl, "RX: Packet size underrun (1)\n");
00706                 goto drop;
00707         }
00708         plcp = (struct b43_plcp_hdr6 *)(skb->data + padding);
00709         skb_pull(skb, sizeof(struct b43_plcp_hdr6) + padding);
00710         /* The skb contains the Wireless Header + payload data now */
00711         if (unlikely(skb->len < (2 + 2 + 6 /*minimum hdr */  + FCS_LEN))) {
00712                 b43dbg(dev->wl, "RX: Packet size underrun (2)\n");
00713                 goto drop;
00714         }
00715         wlhdr = (struct ieee80211_hdr *)(skb->data);
00716         fctl = wlhdr->frame_control;
00717 
00718         if (macstat & B43_RX_MAC_DEC) {
00719                 unsigned int keyidx;
00720                 int wlhdr_len;
00721 
00722                 keyidx = ((macstat & B43_RX_MAC_KEYIDX)
00723                           >> B43_RX_MAC_KEYIDX_SHIFT);
00724                 /* We must adjust the key index here. We want the "physical"
00725                  * key index, but the ucode passed it slightly different.
00726                  */
00727                 keyidx = b43_kidx_to_raw(dev, keyidx);
00728                 B43_WARN_ON(keyidx >= ARRAY_SIZE(dev->key));

00735                                 goto drop;

00739         }

00746                         max(rxhdr->phy_ht_power0, rxhdr->phy_ht_power1),
00747                         rxhdr->phy_ht_power2);

00751                 if (rxhdr->power0 == 16 || rxhdr->power0 == 32)
00752                         status.signal = max(rxhdr->power1, rxhdr->power2);
00753                 else
00754                         status.signal = max(rxhdr->power0, rxhdr->power1);
00755                 break;
00756         case B43_PHYTYPE_A:
00757         case B43_PHYTYPE_B:
00758         case B43_PHYTYPE_G:
00759         case B43_PHYTYPE_LP:
00760                 status.signal = b43_rssi_postprocess(dev, rxhdr->jssi,
00761                                                   (phystat0 & B43_RX_PHYST0_OFDM),
00762                                                   (phystat0 & B43_RX_PHYST0_GAINCTL),
00763                                                   (phystat3 & B43_RX_PHYST3_TRSTATE));
00764                 break;
00765         }
00766 
00767         if (phystat0 & B43_RX_PHYST0_OFDM)

00769                                                 phytype == B43_PHYTYPE_A);
00770         else

00870 /* Fill out the mac80211 TXstatus report based on the b43-specific
00871  * txstatus report data. This returns a boolean whether the frame was
00872  * successfully transmitted. */
00873 bool b43_fill_txstatus_report(struct b43_wldev *dev,
00874                               struct ieee80211_tx_info *report,
00875                               const struct b43_txstatus *status)
00876 {
00877         bool frame_success = 1;
00878         int retry_limit;
00879 
00880         /* preserve the confiured retry limit before clearing the status
00881          * The xmit function has overwritten the rc's value with the actual
00882          * retry limit done by the hardware */
00883         retry_limit = report->status.rates[0].count;
00884         ieee80211_tx_info_clear_status(report);
00885 
00886         if (status->acked) {
00887                 /* The frame was ACKed. */
00888                 report->flags |= IEEE80211_TX_STAT_ACK;
00889         } else {
00890                 /* The frame was not ACKed... */
00891                 if (!(report->flags & IEEE80211_TX_CTL_NO_ACK)) {
00892                         /* ...but we expected an ACK. */
00893                         frame_success = 0;
00894                 }
00895         }
00896         if (status->frame_count == 0) {
00897                 /* The frame was not transmitted at all. */
00898                 report->status.rates[0].count = 0;
00899         } else if (status->rts_count > dev->wl->hw->conf.short_frame_max_tx_count) {
00900                 /*
00901                  * If the short retries (RTS, not data frame) have exceeded
00902                  * the limit, the hw will not have tried the selected rate,
00903                  * but will have used the fallback rate instead.
00904                  * Don't let the rate control count attempts for the selected
00905                  * rate in this case, otherwise the statistics will be off.
00906                  */
00907                 report->status.rates[0].count = 0;
00908                 report->status.rates[1].count = status->frame_count;
00909         } else {
00910                 if (status->frame_count > retry_limit) {
00911                         report->status.rates[0].count = retry_limit;
00912                         report->status.rates[1].count = status->frame_count -
00913                                         retry_limit;
00914 
00915                 } else {
00916                         report->status.rates[0].count = status->frame_count;
00917                         report->status.rates[1].idx = -1;
00918                 }
00919         }
00920 
00921         return frame_success;
00922 }
