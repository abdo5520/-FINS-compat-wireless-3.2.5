02823 {
02824         u32 type, value;
02825 
02826         type = offset & B43_NTAB_TYPEMASK;
02827         offset &= ~B43_NTAB_TYPEMASK;
02828         B43_WARN_ON(offset > 0xFFFF);
02829 
02830         switch (type) {

02832                 b43_phy_write(dev, B43_NPHY_TABLE_ADDR, offset);
02833                 value = b43_phy_read(dev, B43_NPHY_TABLE_DATALO) & 0xFF;
02834                 break;
02835         case B43_NTAB_16BIT:
02836                 b43_phy_write(dev, B43_NPHY_TABLE_ADDR, offset);
02837                 value = b43_phy_read(dev, B43_NPHY_TABLE_DATALO);

02840                 b43_phy_write(dev, B43_NPHY_TABLE_ADDR, offset);
02841                 value = b43_phy_read(dev, B43_NPHY_TABLE_DATAHI);
02842                 value <<= 16;
02843                 value |= b43_phy_read(dev, B43_NPHY_TABLE_DATALO);
02844                 break;
02845         default:
02846                 B43_WARN_ON(1);
02847                 value = 0;
02848         }
