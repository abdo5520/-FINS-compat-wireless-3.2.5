00050 static u32 b43_dma_address(struct b43_dma *dma, dma_addr_t dmaaddr,
00051                            enum b43_addrtype addrtype)
00052 {
00053         u32 uninitialized_var(addr);
00054 
00055         switch (addrtype) {
00056         case B43_DMA_ADDR_LOW:
00057                 addr = lower_32_bits(dmaaddr);
00058                 if (dma->translation_in_low) {
00059                         addr &= ~SSB_DMA_TRANSLATION_MASK;
00060                         addr |= dma->translation;
00061                 }
00062                 break;
00063         case B43_DMA_ADDR_HIGH:
00064                 addr = upper_32_bits(dmaaddr);
00065                 if (!dma->translation_in_low) {
00066                         addr &= ~SSB_DMA_TRANSLATION_MASK;
00067                         addr |= dma->translation;
00068                 }
00069                 break;
00070         case B43_DMA_ADDR_EXT:
00071                 if (dma->translation_in_low)
00072                         addr = lower_32_bits(dmaaddr);
00073                 else
00074                         addr = upper_32_bits(dmaaddr);
00075                 addr &= SSB_DMA_TRANSLATION_MASK;
00076                 addr >>= SSB_DMA_TRANSLATION_SHIFT;
00077                 break;
00078         }
00079 
00080         return addr;
00081 }

00815 static u64 supported_dma_mask(struct b43_wldev *dev)
00816 {
00817         u32 tmp;

00819 
00820         switch (dev->dev->bus_type) {
00821 #ifdef CONFIG_B43_BCMA
00822         case B43_BUS_BCMA:

00840         if (tmp & B43_DMA32_TXADDREXT_MASK)
00841                 return DMA_BIT_MASK(32);

01061 static int b43_dma_set_mask(struct b43_wldev *dev, u64 mask)
01062 {
01063         u64 orig_mask = mask;
01064         bool fallback = 0;
01065         int err;
01066 
01067         /* Try to set the DMA mask. If it fails, try falling back to a
01068          * lower mask, as we can always also support a lower one. */
01069         while (1) {
01070                 err = dma_set_mask(dev->dev->dma_dev, mask);
01071                 if (!err) {
01072                         err = dma_set_coherent_mask(dev->dev->dma_dev, mask);
01073                         if (!err)
01074                                 break;
01075                 }
01076                 if (mask == DMA_BIT_MASK(64)) {
01077                         mask = DMA_BIT_MASK(32);
01078                         fallback = 1;
01079                         continue;
01080                 }
01081                 if (mask == DMA_BIT_MASK(32)) {
01082                         mask = DMA_BIT_MASK(30);
01083                         fallback = 1;
01084                         continue;
01085                 }

01087                        "the required %u-bit DMA mask\n",

01089                 return -EOPNOTSUPP;

01092                 b43info(dev->wl, "DMA mask fallback from %u-bit to %u-bit\n",
01093                         (unsigned int)dma_mask_to_engine_type(orig_mask),

01167         dma->tx_ring_AC_VO = b43_setup_dmaring(dev, 3, 1, type);
01168         if (!dma->tx_ring_AC_VO)
01169                 goto err_destroy_vi;
01170 
01171         dma->tx_ring_mcast = b43_setup_dmaring(dev, 4, 1, type);
01172         if (!dma->tx_ring_mcast)
01173                 goto err_destroy_vo;
01174 
01175         /* setup RX DMA channel. */
01176         dma->rx_ring = b43_setup_dmaring(dev, 0, 0, type);
01177         if (!dma->rx_ring)
01178                 goto err_destroy_mcast;
01179 
01180         /* No support for the TX status DMA ring. */
01181         B43_WARN_ON(dev->dev->core_rev < 5);
01182 
01183         b43dbg(dev->wl, "%u-bit DMA initialized\n",
01184                (unsigned int)type);
01185         err = 0;
01186 out:
01187         return err;
01188 
01189 err_destroy_mcast:
01190         destroy_ring(dma, tx_ring_mcast);
01191 err_destroy_vo:
01192         destroy_ring(dma, tx_ring_AC_VO);
01193 err_destroy_vi:
01194         destroy_ring(dma, tx_ring_AC_VI);
01195 err_destroy_be:
01196         destroy_ring(dma, tx_ring_AC_BE);
01197 err_destroy_bk:
01198         destroy_ring(dma, tx_ring_AC_BK);
01199         return err;
01200 }

01202 /* Generate a cookie for the TX header. */
01203 static u16 generate_cookie(struct b43_dmaring *ring, int slot)
01204 {
01205         u16 cookie;
01206 
01207         /* Use the upper 4 bits of the cookie as
01208          * DMA controller ID and store the slot number
01209          * in the lower 12 bits.
01210          * Note that the cookie must never be 0, as this
01211          * is a special value used in RX path.
01212          * It can also not be 0xFFFF because that is special
01213          * for multicast frames.
01214          */
01215         cookie = (((u16)ring->index + 1) << 12);
01216         B43_WARN_ON(slot & ~0x0FFF);
01217         cookie |= (u16)slot;
01218 
01219         return cookie;
01220 }

01222 /* Inspect a cookie and find out to which controller/slot it belongs. */
01223 static
01224 struct b43_dmaring *parse_cookie(struct b43_wldev *dev, u16 cookie, int *slot)
01225 {
01226         struct b43_dma *dma = &dev->dma;
01227         struct b43_dmaring *ring = NULL;
01228 
01229         switch (cookie & 0xF000) {
01230         case 0x1000:
01231                 ring = dma->tx_ring_AC_BK;
01232                 break;
01233         case 0x2000:
01234                 ring = dma->tx_ring_AC_BE;
01235                 break;
01236         case 0x3000:
01237                 ring = dma->tx_ring_AC_VI;
01238                 break;
01239         case 0x4000:
01240                 ring = dma->tx_ring_AC_VO;
01241                 break;
01242         case 0x5000:
01243                 ring = dma->tx_ring_mcast;
01244                 break;
01245         }
01246         *slot = (cookie & 0x0FFF);
01247         if (unlikely(!ring || *slot < 0 || *slot >= ring->nr_slots)) {
01248                 b43dbg(dev->wl, "TX-status contains "
01249                        "invalid cookie: 0x%04X\n", cookie);
01250                 return NULL;
01251         }
01252 
01253         return ring;
01254 }

01256 static int dma_tx_fragment(struct b43_dmaring *ring,
01257                            struct sk_buff *skb)
01258 {
01259         const struct b43_dma_ops *ops = ring->ops;
01260         struct ieee80211_tx_info *info = IEEE80211_SKB_CB(skb);
01261         struct b43_private_tx_info *priv_info = b43_get_priv_tx_info(info);
01262         u8 *header;
01263         int slot, old_top_slot, old_used_slots;
01264         int err;
01265         struct b43_dmadesc_generic *desc;
01266         struct b43_dmadesc_meta *meta;
01267         struct b43_dmadesc_meta *meta_hdr;
01268         u16 cookie;
01269         size_t hdrsize = b43_txhdr_size(ring->dev);
01270 
01271         /* Important note: If the number of used DMA slots per TX frame
01272          * is changed here, the TX_SLOTS_PER_FRAME definition at the top of
01273          * the file has to be updated, too!
01274          */
01275 
01276         old_top_slot = ring->current_slot;
01277         old_used_slots = ring->used_slots;
01278 
01279         /* Get a slot for the header. */
01280         slot = request_slot(ring);
01281         desc = ops->idx2desc(ring, slot, &meta_hdr);
01282         memset(meta_hdr, 0, sizeof(*meta_hdr));
01283 
01284         header = &(ring->txhdr_cache[(slot / TX_SLOTS_PER_FRAME) * hdrsize]);
01285         cookie = generate_cookie(ring, slot);
01286         err = b43_generate_txhdr(ring->dev, header,
01287                                  skb, info, cookie);
01288         if (unlikely(err)) {
01289                 ring->current_slot = old_top_slot;
01290                 ring->used_slots = old_used_slots;
01291                 return err;
01292         }
01293 
01294         meta_hdr->dmaaddr = map_descbuffer(ring, (unsigned char *)header,
01295                                            hdrsize, 1);
01296         if (b43_dma_mapping_error(ring, meta_hdr->dmaaddr, hdrsize, 1)) {
01297                 ring->current_slot = old_top_slot;
01298                 ring->used_slots = old_used_slots;
01299                 return -EIO;
01300         }
01301         ops->fill_descriptor(ring, desc, meta_hdr->dmaaddr,
01302                              hdrsize, 1, 0, 0);
01303 
01304         /* Get a slot for the payload. */
01305         slot = request_slot(ring);
01306         desc = ops->idx2desc(ring, slot, &meta);
01307         memset(meta, 0, sizeof(*meta));
01308 
01309         meta->skb = skb;
01310         meta->is_last_fragment = 1;
01311         priv_info->bouncebuffer = NULL;
01312 
01313         meta->dmaaddr = map_descbuffer(ring, skb->data, skb->len, 1);
01314         /* create a bounce buffer in zone_dma on mapping failure. */
01315         if (b43_dma_mapping_error(ring, meta->dmaaddr, skb->len, 1)) {
01316                 priv_info->bouncebuffer = kmemdup(skb->data, skb->len,
01317                                                   GFP_ATOMIC | GFP_DMA);
01318                 if (!priv_info->bouncebuffer) {
01319                         ring->current_slot = old_top_slot;
01320                         ring->used_slots = old_used_slots;
01321                         err = -ENOMEM;
01322                         goto out_unmap_hdr;
01323                 }
01324 
01325                 meta->dmaaddr = map_descbuffer(ring, priv_info->bouncebuffer, skb->len, 1);
01326                 if (b43_dma_mapping_error(ring, meta->dmaaddr, skb->len, 1)) {
01327                         kfree(priv_info->bouncebuffer);
01328                         priv_info->bouncebuffer = NULL;
01329                         ring->current_slot = old_top_slot;
01330                         ring->used_slots = old_used_slots;
01331                         err = -EIO;
01332                         goto out_unmap_hdr;
01333                 }
01334         }
01335 
01336         ops->fill_descriptor(ring, desc, meta->dmaaddr, skb->len, 0, 1, 1);
01337 
01338         if (info->flags & IEEE80211_TX_CTL_SEND_AFTER_DTIM) {
01339                 /* Tell the firmware about the cookie of the last
01340                  * mcast frame, so it can clear the more-data bit in it. */
01341                 b43_shm_write16(ring->dev, B43_SHM_SHARED,
01342                                 B43_SHM_SH_MCASTCOOKIE, cookie);
01343         }
01344         /* Now transfer the whole frame. */
01345         wmb();
01346         ops->poke_tx(ring, next_slot(ring, slot));
01347         return 0;
01348 
01349 out_unmap_hdr:
01350         unmap_descbuffer(ring, meta_hdr->dmaaddr,
01351                          hdrsize, 1);
01352         return err;
01353 }

01378                                                    u8 queue_prio)

01381 
01382         if (dev->qos_enabled) {
01383                 /* 0 = highest priority */
01384                 switch (queue_prio) {
01385                 default:
01386                         B43_WARN_ON(1);
01387                         /* fallthrough */
01388                 case 0:

01390                         break;
01391                 case 1:
01392                         ring = dev->dma.tx_ring_AC_VI;
01393                         break;
01394                 case 2:
01395                         ring = dev->dma.tx_ring_AC_BE;
01396                         break;
01397                 case 3:
01398                         ring = dev->dma.tx_ring_AC_BK;
01399                         break;
01400                 }
01401         } else
01402                 ring = dev->dma.tx_ring_AC_BE;

01407 int b43_dma_tx(struct b43_wldev *dev, struct sk_buff *skb)
01408 {
01409         struct b43_dmaring *ring;
01410         struct ieee80211_hdr *hdr;
01411         int err = 0;
01412         struct ieee80211_tx_info *info = IEEE80211_SKB_CB(skb);
01413 
01414         hdr = (struct ieee80211_hdr *)skb->data;
01415         if (info->flags & IEEE80211_TX_CTL_SEND_AFTER_DTIM) {
01416                 /* The multicast ring will be sent after the DTIM */
01417                 ring = dev->dma.tx_ring_mcast;
01418                 /* Set the more-data bit. Ucode will clear it on
01419                  * the last frame for us. */
01420                 hdr->frame_control |= cpu_to_le16(IEEE80211_FCTL_MOREDATA);
01421         } else {
01422                 /* Decide by priority where to put this frame. */
01423                 ring = select_ring_by_priority(
01424                         dev, skb_get_queue_mapping(skb));
01425         }
01426 
01427         B43_WARN_ON(!ring->tx);
01428 
01429         if (unlikely(ring->stopped)) {
01430                 /* We get here only because of a bug in mac80211.
01431                  * Because of a race, one packet may be queued after
01432                  * the queue is stopped, thus we got called when we shouldn't.
01433                  * For now, just refuse the transmit. */
01434                 if (b43_debug(dev, B43_DBG_DMAVERBOSE))
01435                         b43err(dev->wl, "Packet after queue stopped\n");
01436                 err = -ENOSPC;
01437                 goto out;
01438         }
01439 
01440         if (unlikely(WARN_ON(free_slots(ring) < TX_SLOTS_PER_FRAME))) {
01441                 /* If we get here, we have a real error with the queue
01442                  * full, but queues not stopped. */
01443                 b43err(dev->wl, "DMA queue overflow\n");
01444                 err = -ENOSPC;
01445                 goto out;
01446         }
01447 
01448         /* Assign the queue number to the ring (if not already done before)
01449          * so TX status handling can use it. The queue to ring mapping is
01450          * static, so we don't need to store it per frame. */
01451         ring->queue_prio = skb_get_queue_mapping(skb);
01452 
01453         err = dma_tx_fragment(ring, skb);
01454         if (unlikely(err == -ENOKEY)) {
01455                 /* Drop this packet, as we don't have the encryption key
01456                  * anymore and must not transmit it unencrypted. */
01457                 dev_kfree_skb_any(skb);
01458                 err = 0;
01459                 goto out;
01460         }
01461         if (unlikely(err)) {
01462                 b43err(dev->wl, "DMA tx mapping failure\n");
01463                 goto out;
01464         }
01465         if ((free_slots(ring) < TX_SLOTS_PER_FRAME) ||
01466             should_inject_overflow(ring)) {
01467                 /* This TX ring is full. */
01468                 ieee80211_stop_queue(dev->wl->hw, skb_get_queue_mapping(skb));
01469                 ring->stopped = 1;
01470                 if (b43_debug(dev, B43_DBG_DMAVERBOSE)) {
01471                         b43dbg(dev->wl, "Stopped TX ring %d\n", ring->index);
01472                 }
01473         }
01474 out:
01475 
01476         return err;
01477 }

01479 void b43_dma_handle_txstatus(struct b43_wldev *dev,
01480                              const struct b43_txstatus *status)
01481 {
01482         const struct b43_dma_ops *ops;
01483         struct b43_dmaring *ring;
01484         struct b43_dmadesc_meta *meta;
01485         int slot, firstused;
01486         bool frame_succeed;
01487 
01488         ring = parse_cookie(dev, status->cookie, &slot);
01489         if (unlikely(!ring))
01490                 return;
01491         B43_WARN_ON(!ring->tx);
01492 
01493         /* Sanity check: TX packets are processed in-order on one ring.
01494          * Check if the slot deduced from the cookie really is the first
01495          * used slot. */
01496         firstused = ring->current_slot - ring->used_slots + 1;
01497         if (firstused < 0)
01498                 firstused = ring->nr_slots + firstused;
01499         if (unlikely(slot != firstused)) {
01500                 /* This possibly is a firmware bug and will result in
01501                  * malfunction, memory leaks and/or stall of DMA functionality. */
01502                 b43dbg(dev->wl, "Out of order TX status report on DMA ring %d. "
01503                        "Expected %d, but got %d\n",
01504                        ring->index, firstused, slot);
01505                 return;
01506         }
01507 
01508         ops = ring->ops;
01509         while (1) {
01510                 B43_WARN_ON(slot < 0 || slot >= ring->nr_slots);
01511                 /* get meta - ignore returned value */
01512                 ops->idx2desc(ring, slot, &meta);
01513 
01514                 if (b43_dma_ptr_is_poisoned(meta->skb)) {
01515                         b43dbg(dev->wl, "Poisoned TX slot %d (first=%d) "
01516                                "on ring %d\n",
01517                                slot, firstused, ring->index);
01518                         break;
01519                 }
01520                 if (meta->skb) {
01521                         struct b43_private_tx_info *priv_info =
01522                                 b43_get_priv_tx_info(IEEE80211_SKB_CB(meta->skb));
01523 
01524                         unmap_descbuffer(ring, meta->dmaaddr, meta->skb->len, 1);
01525                         kfree(priv_info->bouncebuffer);
01526                         priv_info->bouncebuffer = NULL;
01527                 } else {
01528                         unmap_descbuffer(ring, meta->dmaaddr,
01529                                          b43_txhdr_size(dev), 1);
01530                 }
01531 
01532                 if (meta->is_last_fragment) {
01533                         struct ieee80211_tx_info *info;
01534 
01535                         if (unlikely(!meta->skb)) {
01536                                 /* This is a scatter-gather fragment of a frame, so
01537                                  * the skb pointer must not be NULL. */
01538                                 b43dbg(dev->wl, "TX status unexpected NULL skb "
01539                                        "at slot %d (first=%d) on ring %d\n",
01540                                        slot, firstused, ring->index);
01541                                 break;
01542                         }
01543 
01544                         info = IEEE80211_SKB_CB(meta->skb);
01545 
01546                         /*
01547                          * Call back to inform the ieee80211 subsystem about
01548                          * the status of the transmission.
01549                          */
01550                         frame_succeed = b43_fill_txstatus_report(dev, info, status);
01551 #ifdef CONFIG_B43_DEBUG
01552                         if (frame_succeed)
01553                                 ring->nr_succeed_tx_packets++;
01554                         else
01555                                 ring->nr_failed_tx_packets++;
01556                         ring->nr_total_packet_tries += status->frame_count;
01557 #endif /* DEBUG */
01558                         ieee80211_tx_status(dev->wl->hw, meta->skb);
01559 
01560                         /* skb will be freed by ieee80211_tx_status().
01561                          * Poison our pointer. */
01562                         meta->skb = B43_DMA_PTR_POISON;
01563                 } else {
01564                         /* No need to call free_descriptor_buffer here, as
01565                          * this is only the txhdr, which is not allocated.
01566                          */
01567                         if (unlikely(meta->skb)) {
01568                                 b43dbg(dev->wl, "TX status unexpected non-NULL skb "
01569                                        "at slot %d (first=%d) on ring %d\n",
01570                                        slot, firstused, ring->index);
01571                                 break;
01572                         }
01573                 }
01574 
01575                 /* Everything unmapped and free'd. So it's not used anymore. */
01576                 ring->used_slots--;
01577 
01578                 if (meta->is_last_fragment) {
01579                         /* This is the last scatter-gather
01580                          * fragment of the frame. We are done. */
01581                         break;
01582                 }
01583                 slot = next_slot(ring, slot);
01584         }
01585         if (ring->stopped) {
01586                 B43_WARN_ON(free_slots(ring) < TX_SLOTS_PER_FRAME);
01587                 ieee80211_wake_queue(dev->wl->hw, ring->queue_prio);
01588                 ring->stopped = 0;
01589                 if (b43_debug(dev, B43_DBG_DMAVERBOSE)) {
01590                         b43dbg(dev->wl, "Woke up TX ring %d\n", ring->index);
01591                 }
01592         }
01593 }

01595 static void dma_rx(struct b43_dmaring *ring, int *slot)
01596 {
01597         const struct b43_dma_ops *ops = ring->ops;
01598         struct b43_dmadesc_generic *desc;
01599         struct b43_dmadesc_meta *meta;
01600         struct b43_rxhdr_fw4 *rxhdr;
01601         struct sk_buff *skb;
01602         u16 len;
01603         int err;
01604         dma_addr_t dmaaddr;
01605 
01606         desc = ops->idx2desc(ring, *slot, &meta);
01607 
01608         sync_descbuffer_for_cpu(ring, meta->dmaaddr, ring->rx_buffersize);
01609         skb = meta->skb;
01610 
01611         rxhdr = (struct b43_rxhdr_fw4 *)skb->data;
01612         len = le16_to_cpu(rxhdr->frame_len);
01613         if (len == 0) {
01614                 int i = 0;
01615 
01616                 do {
01617                         udelay(2);
01618                         barrier();
01619                         len = le16_to_cpu(rxhdr->frame_len);
01620                 } while (len == 0 && i++ < 5);
01621                 if (unlikely(len == 0)) {
01622                         dmaaddr = meta->dmaaddr;
01623                         goto drop_recycle_buffer;
01624                 }
01625         }
01626         if (unlikely(b43_rx_buffer_is_poisoned(ring, skb))) {
01627                 /* Something went wrong with the DMA.
01628                  * The device did not touch the buffer and did not overwrite the poison. */
01629                 b43dbg(ring->dev->wl, "DMA RX: Dropping poisoned buffer.\n");
01630                 dmaaddr = meta->dmaaddr;
01631                 goto drop_recycle_buffer;
01632         }
01633         if (unlikely(len + ring->frameoffset > ring->rx_buffersize)) {
01634                 /* The data did not fit into one descriptor buffer
01635                  * and is split over multiple buffers.
01636                  * This should never happen, as we try to allocate buffers
01637                  * big enough. So simply ignore this packet.
01638                  */
01639                 int cnt = 0;
01640                 s32 tmp = len;
01641 
01642                 while (1) {
01643                         desc = ops->idx2desc(ring, *slot, &meta);
01644                         /* recycle the descriptor buffer. */
01645                         b43_poison_rx_buffer(ring, meta->skb);
01646                         sync_descbuffer_for_device(ring, meta->dmaaddr,
01647                                                    ring->rx_buffersize);
01648                         *slot = next_slot(ring, *slot);
01649                         cnt++;
01650                         tmp -= ring->rx_buffersize;
01651                         if (tmp <= 0)
01652                                 break;
01653                 }
01654                 b43err(ring->dev->wl, "DMA RX buffer too small "
01655                        "(len: %u, buffer: %u, nr-dropped: %d)\n",
01656                        len, ring->rx_buffersize, cnt);
01657                 goto drop;
01658         }
01659 
01660         dmaaddr = meta->dmaaddr;
01661         err = setup_rx_descbuffer(ring, desc, meta, GFP_ATOMIC);
01662         if (unlikely(err)) {
01663                 b43dbg(ring->dev->wl, "DMA RX: setup_rx_descbuffer() failed\n");
01664                 goto drop_recycle_buffer;
01665         }
01666 
01667         unmap_descbuffer(ring, dmaaddr, ring->rx_buffersize, 0);
01668         skb_put(skb, len + ring->frameoffset);
01669         skb_pull(skb, ring->frameoffset);
01670 
01671         b43_rx(ring->dev, skb, rxhdr);
01672 drop:
01673         return;
01674 
01675 drop_recycle_buffer:
01676         /* Poison and recycle the RX buffer. */
01677         b43_poison_rx_buffer(ring, skb);
01678         sync_descbuffer_for_device(ring, dmaaddr, ring->rx_buffersize);
01679 }

01683         const struct b43_dma_ops *ops = ring->ops;
