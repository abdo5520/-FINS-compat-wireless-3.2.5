00059 {

00078 static u16 lo_measure_feedthrough(struct b43_wldev *dev,
00079                                   u16 lna, u16 pga, u16 trsw_rx)
00080 {
00081         struct b43_phy *phy = &dev->phy;
00082         u16 rfover;
00083         u16 feedthrough;
00084 
00085         if (phy->gmode) {
00086                 lna <<= B43_PHY_RFOVERVAL_LNA_SHIFT;
00087                 pga <<= B43_PHY_RFOVERVAL_PGA_SHIFT;
00088 
00089                 B43_WARN_ON(lna & ~B43_PHY_RFOVERVAL_LNA);
00090                 B43_WARN_ON(pga & ~B43_PHY_RFOVERVAL_PGA);
00091 /*FIXME This assertion fails            B43_WARN_ON(trsw_rx & ~(B43_PHY_RFOVERVAL_TRSWRX |
00092                                     B43_PHY_RFOVERVAL_BW));
00093 */
00094                 trsw_rx &= (B43_PHY_RFOVERVAL_TRSWRX | B43_PHY_RFOVERVAL_BW);
00095 
00096                 /* Construct the RF Override Value */
00097                 rfover = B43_PHY_RFOVERVAL_UNK;
00098                 rfover |= pga;
00099                 rfover |= lna;
00100                 rfover |= trsw_rx;
00101                 if ((dev->dev->bus_sprom->boardflags_lo & B43_BFL_EXTLNA)
00102                     && phy->rev > 6)
00103                         rfover |= B43_PHY_RFOVERVAL_EXTLNA;

00105                 b43_phy_write(dev, B43_PHY_PGACTL, 0xE300);
00106                 b43_phy_write(dev, B43_PHY_RFOVERVAL, rfover);
00107                 udelay(10);

00109                 b43_phy_write(dev, B43_PHY_RFOVERVAL, rfover);
00110                 udelay(10);
00111                 rfover |= B43_PHY_RFOVERVAL_BW_LPF;
00112                 b43_phy_write(dev, B43_PHY_RFOVERVAL, rfover);
00113                 udelay(10);

00292 /* 802.11/LO/GPHY/MeasuringGains */
00293 static void lo_measure_gain_values(struct b43_wldev *dev,
00294                                    s16 max_rx_gain, int use_trsw_rx)
00295 {
00296         struct b43_phy *phy = &dev->phy;
00297         struct b43_phy_g *gphy = phy->g;
00298         u16 tmp;
00299 
00300         if (max_rx_gain < 0)
00301                 max_rx_gain = 0;
00302 
00303         if (has_loopback_gain(phy)) {
00304                 int trsw_rx_gain;
00305 
00306                 if (use_trsw_rx) {
00307                         trsw_rx_gain = gphy->trsw_rx_gain / 2;
00308                         if (max_rx_gain >= trsw_rx_gain) {
00309                                 trsw_rx_gain = max_rx_gain - trsw_rx_gain;
00310                         }
00311                 } else
00312                         trsw_rx_gain = max_rx_gain;
00313                 if (trsw_rx_gain < 9) {
00314                         gphy->lna_lod_gain = 0;
00315                 } else {
00316                         gphy->lna_lod_gain = 1;
00317                         trsw_rx_gain -= 8;
00318                 }
00319                 trsw_rx_gain = clamp_val(trsw_rx_gain, 0, 0x2D);
00320                 gphy->pga_gain = trsw_rx_gain / 3;
00321                 if (gphy->pga_gain >= 5) {
00322                         gphy->pga_gain -= 5;
00323                         gphy->lna_gain = 2;
00324                 } else
00325                         gphy->lna_gain = 0;
00326         } else {
00327                 gphy->lna_gain = 0;
00328                 gphy->trsw_rx_gain = 0x20;
00329                 if (max_rx_gain >= 0x14) {
00330                         gphy->lna_lod_gain = 1;
00331                         gphy->pga_gain = 2;
00332                 } else if (max_rx_gain >= 0x12) {
00333                         gphy->lna_lod_gain = 1;
00334                         gphy->pga_gain = 1;
00335                 } else if (max_rx_gain >= 0xF) {
00336                         gphy->lna_lod_gain = 1;
00337                         gphy->pga_gain = 0;
00338                 } else {
00339                         gphy->lna_lod_gain = 0;
00340                         gphy->pga_gain = 0;
00341                 }
00342         }
00343 
00344         tmp = b43_radio_read16(dev, 0x7A);
00345         if (gphy->lna_lod_gain == 0)
00346                 tmp &= ~0x0008;
00347         else
00348                 tmp |= 0x0008;
00349         b43_radio_write16(dev, 0x7A, tmp);
00350 }

00819 void b43_gphy_dc_lt_init(struct b43_wldev *dev, bool update_all)
00820 {
00821         struct b43_phy *phy = &dev->phy;
00822         struct b43_phy_g *gphy = phy->g;
00823         struct b43_txpower_lo_control *lo = gphy->lo_control;
00824         int i;
00825         int rf_offset, bb_offset;
00826         const struct b43_rfatt *rfatt;
00827         const struct b43_bbatt *bbatt;
00828         u64 power_vector;
00829         bool table_changed = 0;
00830 
00831         BUILD_BUG_ON(B43_DC_LT_SIZE != 32);
00832         B43_WARN_ON(lo->rfatt_list.len * lo->bbatt_list.len > 64);
00833 
00834         power_vector = lo->power_vector;
00835         if (!update_all && !power_vector)
00836                 return; /* Nothing to do. */
00837 
00838         /* Suspend the MAC now to avoid continuous suspend/enable
00839          * cycles in the loop. */
00840         b43_mac_suspend(dev);
00841 
00842         for (i = 0; i < B43_DC_LT_SIZE * 2; i++) {
00843                 struct b43_lo_calib *cal;
00844                 int idx;
00845                 u16 val;
00846 
00847                 if (!update_all && !(power_vector & (((u64)1ULL) << i)))
00848                         continue;
00849                 /* Update the table entry for this power_vector bit.
00850                  * The table rows are RFatt entries and columns are BBatt. */
00851                 bb_offset = i / lo->rfatt_list.len;
00852                 rf_offset = i % lo->rfatt_list.len;
00853                 bbatt = &(lo->bbatt_list.list[bb_offset]);
00854                 rfatt = &(lo->rfatt_list.list[rf_offset]);
00855 
00856                 cal = b43_calibrate_lo_setting(dev, bbatt, rfatt);
00857                 if (!cal) {
00858                         b43warn(dev->wl, "LO: Could not "
00859                                 "calibrate DC table entry\n");
00860                         continue;
00861                 }
00862                 /*FIXME: Is Q really in the low nibble? */
00863                 val = (u8)(cal->ctl.q);
00864                 val |= ((u8)(cal->ctl.i)) << 4;
00865                 kfree(cal);
00866 
00867                 /* Get the index into the hardware DC LT. */
00868                 idx = i / 2;
00869                 /* Change the table in memory. */
00870                 if (i % 2) {
00871                         /* Change the high byte. */
00872                         lo->dc_lt[idx] = (lo->dc_lt[idx] & 0x00FF)
00873                                          | ((val & 0x00FF) << 8);
00874                 } else {
00875                         /* Change the low byte. */
00876                         lo->dc_lt[idx] = (lo->dc_lt[idx] & 0xFF00)
00877                                          | (val & 0x00FF);
00878                 }
00879                 table_changed = 1;

00882                 /* The table changed in memory. Update the hardware table. */
00883                 for (i = 0; i < B43_DC_LT_SIZE; i++)
