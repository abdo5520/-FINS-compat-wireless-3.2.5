00420 static unsigned long calc_expire_secs(unsigned long now,
00421                                       unsigned long time,
00422                                       unsigned long expire)
00423 {
00424         expire = time + expire;
00425 
00426         if (time_after(now, expire))
00427                 return 0; /* expired */
00428         if (expire < now) {
00429                 /* jiffies wrapped */
00430                 expire -= MAX_JIFFY_OFFSET;
00431                 now -= MAX_JIFFY_OFFSET;
00432         }
00433         B43_WARN_ON(expire < now);
00434 
00435         return (expire - now) / HZ;
00436 }

00509 static ssize_t b43_debugfs_read(struct file *file, char __user *userbuf,
00510                                 size_t count, loff_t *ppos)
00511 {
00512         struct b43_wldev *dev;
00513         struct b43_debugfs_fops *dfops;
00514         struct b43_dfs_file *dfile;
00515         ssize_t uninitialized_var(ret);
00516         char *buf;
00517         const size_t bufsize = 1024 * 16; /* 16 kiB buffer */
00518         const size_t buforder = get_order(bufsize);
00519         int err = 0;
00520 
00521         if (!count)
00522                 return 0;
00523         dev = file->private_data;
00524         if (!dev)
00525                 return -ENODEV;
00526 
00527         mutex_lock(&dev->wl->mutex);
00528         if (b43_status(dev) < B43_STAT_INITIALIZED) {
00529                 err = -ENODEV;
00530                 goto out_unlock;
00531         }
00532 
00533         dfops = container_of(file->f_op, struct b43_debugfs_fops, fops);
00534         if (!dfops->read) {
00535                 err = -ENOSYS;
00536                 goto out_unlock;
00537         }
00538         dfile = fops_to_dfs_file(dev, dfops);
00539 
00540         if (!dfile->buffer) {
00541                 buf = (char *)__get_free_pages(GFP_KERNEL, buforder);
00542                 if (!buf) {
00543                         err = -ENOMEM;
00544                         goto out_unlock;
00545                 }
00546                 memset(buf, 0, bufsize);
00547                 ret = dfops->read(dev, buf, bufsize);
00548                 if (ret <= 0) {
00549                         free_pages((unsigned long)buf, buforder);
00550                         err = ret;
00551                         goto out_unlock;
00552                 }
00553                 dfile->data_len = ret;
00554                 dfile->buffer = buf;
00555         }
00556 
00557         ret = simple_read_from_buffer(userbuf, count, ppos,
00558                                       dfile->buffer,
00559                                       dfile->data_len);
00560         if (*ppos >= dfile->data_len) {
00561                 free_pages((unsigned long)dfile->buffer, buforder);
00562                 dfile->buffer = NULL;
00563                 dfile->data_len = 0;
00564         }
00565 out_unlock:
00566         mutex_unlock(&dev->wl->mutex);
00567 
00568         return err ? err : ret;
00569 }
